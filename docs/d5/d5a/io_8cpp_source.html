472 //! Isotopics from time points p satisfying pstart<=p<=pend are saved.
473 //! Consider isotopics data at 5 time points.
474 //!
475 //! p=0 1 2 3 4
476 //! | | | | |
477 //!
478 //! To ask for a dump of p=2,3,4 isotopics, supply pstart=2,pend=4 and the
479 //! * time points below will be dumped.
480 //!
481 //! p=0 1 2 3 4
482 //! | | * * *
483 void saveStateSet( size_t pstart,
484  size_t pend,
485  const MaterialMap& matmap,
486  const std::string& path,
487  ScaleUtils::IO::DB opts )
488 {
489  // BAD_SIZE_T means start at 0
490  size_t local_pstart = pstart == BAD_SIZE_T ? 0 : pstart;
491 
492  // append to existing files if pstart>0 AND not already set (acquire
493  // is a good way to do that)
494  opts.acquire<bool>( "appendExisting", local_pstart > 0 );
495 
496  // add data here
497  StateSet state_set;
498 
499  // iterate through and add data
500  for( const auto& id_mat : matmap )
501  {
502  int id = id_mat.first;
503  const Material& mat = *id_mat.second;
504 
505  // BAD_SIZE_T means go to the end
506  size_t local_pend = pend == BAD_SIZE_T ? mat.ntimes() - 1 : pend;
507  Check( local_pstart <= local_pend );
508  Check( local_pend < mat.ntimes() );
509  for( size_t p = local_pstart; p <= local_pend; ++p )
510  {
511  // save data on the interval i
512  // first map from time point "p" to interval "i"
513  // p=0,1 get the same interval data
514  size_t i = p == 0 ? 0 : p - 1;
515 
516  // populate definition
517  StateInfo def;
518  def.set_power( mat.power_over( i ) );
519  def.set_flux( mat.flux_over( i ) );
520  def.set_step_number( i );
521  def.set_case_number( id );
522  def.set_time( mat.time_at( p ) );
523  def.set_library_position( i );
524 
525  // populate tags
526  TagManager tagman;
527  tagman.setIdTag( "name", mat.name() );
528  tagman.setIdTag( "id", std::to_string( id ) );
529  def.set_tag_manager( tagman );
530 
531  // create state
532  State state;
533 
534  // 1. add definition
535  state.set_definition( def );
536 
537  // 2. add concs
538  Concentrations concs;
539  concs.set_nuclide_set( mat.library()->definition().nuclide_set() );
540  mat.get_concs_at( &concs, p );
541  state.set_concs( concs );
542 
543  // 3. add state
544  state_set.add_states( state );
545  }
546  }
547 
548  // dump data
549  StateSetIO io;
550  io.save( state_set, path, opts );

553 //! saves a range of transition data in a MaterialMap to an FT33 (Library)
554 //!
555 //! Data from time intervals satisfying pstart<=i<pend are saved, pstart
556 //! and pend are the starting and ending time points, and i is the time
557 //! interval. Consider data with 5 time points and therefore 4 intervals.
558 //!
559 //! p=0 1 2 3 4
560 //! | | | | |
561 //! i=0 1 2 3
562 //!
563 //! To ask for a dump of interval i=2,3 data, you would supply pstart=2,pend=4
564 //! and the * intervals below will be dumped.
565 //!
566 //! p=0 1 2 3 4
567 //! | | | * | * |
568 //! i=0 1 2 3
569 //!
570 //! Note that transition data is calculated as an average over an interval
571 //! so that for any data to be dumped, one must have pstart<pend.
572 //!
573 //! Unlike saveStateSet, saveLibraries outputs a file for each material
574 //! individually. The pathfmt argument allows substitution of both the
575 //! Material::name() and Material::id(). The path is written using:
576 //! std::string path=findAndReplace(pathfmt,"$name",mat.name());
577 //! if( path.find("%")!=std::string::npos ){
578 //! path=nprintf(path.c_str(),mat.id());
579 //! }
580 //!
581 void saveLibraries( size_t pstart,
582  size_t pend,
583  const MaterialMap& matmap,
584  const std::string& pathfmt,
585  ScaleUtils::IO::DB opts )
586 {
587  // BAD_SIZE_T means start at 0
588  size_t local_pstart = pstart == BAD_SIZE_T ? 0 : pstart;
589 
590  // append to existing files if pstart>0 AND not already set (acquire
591  // is a good way to do that)
592  opts.acquire<bool>( "appendExisting", local_pstart > 0 );
593 
594  // iterate through and add data
595  for( const auto& id_mat : matmap )
596  {
597  int id = id_mat.first;
598  const Material& mat = *id_mat.second;
599 
600  // BAD_SIZE_T means go to the end
601  size_t local_pend = pend == BAD_SIZE_T ? mat.ntimes() - 1 : pend;
602 
603  // make sure we have something to do
604  if( local_pstart < local_pend )
605  {
606  // create path
607  std::string path =
608  ScaleUtils::IO::findAndReplace( pathfmt, "$name", mat.name() );
609  if( path.find( "%" ) != std::string::npos )
610  {
611  path = nprintf( path.c_str(), id );
612  }
613 
614  // create a library but erase the coefficients
615  //(don't worry, this is really cheap because of COW)
616  Library lib( *mat.library() );
617  lib.resize_transition_coeff( 0 );
618 
619  // add data to library
620  for( size_t i = local_pstart; i < local_pend; ++i )
621  {
622  lib.add_transition_matrix( *mat.transition_matrix_over( i ) );
623  }
624 
625  // dump library
626  LibraryIO io;
627  io.save( lib, path, opts );
628  }
