348 {
349  Require( ids != nullptr );
350 
351  // first sort by zai (and s blocks)
352  std::sort( ids->begin(), ids->end(), ScaleData::compare_nuclide_id_zai() );
353 
354  // remove duplicates
355  ids->erase( std::unique( ids->begin(), ids->end() ), ids->end() );
356

373  }
374

467  * \brief Construct a map comparing nuclides on this set with another set.
468  *
469  * \param a NuclideSet to compare to.
470  * \return A map containing the results.
471  *
472  * The resulting map will contain pairs that have the same IZA in the first and
473  * second position on if the nuclide exists on both sets. If it exists on

476  */
477 std::map<int, int> NuclideSet::compare_nucl_lists(
478  const NuclideSet& a,
479  std::map<int, int>& loc_a,
480  std::map<int, int>& loc_b ) const
481 {
482  std::vector<int> nucl_on_both;
483  std::vector<int> nucl_on_a_only;
484  std::vector<int> nucl_on_b_only;
485 
486  int ni = 0;
487  for( auto i : *b_ids )
488  {
489  auto loc = std::find( a.b_ids->begin(), a.b_ids->end(), i );
490  loc_a.insert( std::pair<int, int>( i, ni++ ) );
491  if( loc == a.b_ids->end() )
492  {
493  nucl_on_a_only.push_back( i );
494  }
495  else
496  {
497  nucl_on_both.push_back( i );
498  }
499  }
500 
501  ni = 0;
502  for( auto i : *( a.b_ids ) )
503  {
504  auto loc = std::find( b_ids->begin(), b_ids->end(), i );
505  loc_b.insert( std::pair<int, int>( i, ni++ ) );
506  if( loc == b_ids->end() )
507  {
508  nucl_on_b_only.push_back( i );
509  }
510  }
511 
512  std::map<int, int> nucl_map;
513 
514  for( auto i : nucl_on_both )
515  {
516  nucl_map.insert( std::pair<int, int>( i, i ) );
517  }
518  for( auto i : nucl_on_a_only )
519  {
520  nucl_map.insert( std::pair<int, int>( i, -1 ) );
521  }
522  for( auto i : nucl_on_b_only )

525  }
526 
527  return nucl_map;
528 }

611 bool NuclideSet::sublib_search_standard( const std::vector<int>& sublibs0,
612  size_t i,

614  SCP_IdMap master_id_map,
615  int* master_to_user,
616  int offset,
617  int id )
618 {

620 
621  // construct the order in which to search sublibs
622  std::deque<int> sublibs( sublibs0.begin(), sublibs0.end() );
623 
624  // special clause to make sure that really light nuclides get associated to
625  // light elements first and not as tertiary fission products or alpha
626  // byproducts
627  // for the actinides, e.g. Li-6 must be associated to a light nuclide
628  if( id < 26000 )

631  if( it != sublibs.end() )

634  sublibs.push_front( SUBLIB_1LT );
635  }
636  }

641  {
642  int test = s * ScaleData::Utils::PIZZZAAA_P_OFFSET + id;
643  IdMap::const_iterator it = master_id_map->find( test );
644  if( it != master_id_map->end() )
645  {
646  if( first )

650  jDebugLine( " searched and placed in sublib s=" << s );

718  SErrLine( 2,

720  << " does not exist in sublib s="
721  << s );
722  all_found = false;
723  }
724  }
725  // s!=0 means we look up directly the nuclide in sublib s
726  else if( s != 0 )
727  {
728  IdMap::const_iterator it = master_id_map->find( id );
729  if( it != master_id_map->end() )
730  {
731  user_to_master[i] = it->second + offset;

733  jDebugLine( " directly placed" );
734  }
735  else

741  all_found = false;
742  }
743  }
744

749  i,
750  user_to_master,

753  offset,

762  }
763  }
764  }
