73 float explicit_calc_og_xs( const int search_iza,
74  const int search_mt,
75  const AmpxLibrary& nxs,
76  const std::vector<float>& flux )
77 {
78  SCOPED_TIMER( "tst.explicit_calc_og_xs" );
79 
80  double xs1 = 0;
81  double sum_flux = 0;
82  for( size_t g = 0; g < flux.size(); ++g )
83  {
84  sum_flux += flux[g];
85  }
86  int num_ampx_nuclides = nxs.getNumberNuclides();
87  for( int j = 0; j < num_ampx_nuclides; j++ )
88  {
89  LibraryNuclide* ampx_nuclide = nxs.getNuclideAt( j );
90  int iza = ampx_nuclide->getId();
91  if( iza != search_iza ) continue;
92  int num_1dxs = ampx_nuclide->getNumNeutron1dData();
93  for( int k = 0; k < num_1dxs; k++ )
94  {
95  CrossSection1d* xs = ampx_nuclide->getNeutron1dDataAt( k );
96  int mt = xs->getMt();
97  bool partial;
98  int final_isomeric_state, initial_isomeric_state;
99  ScaleData::Utils::transform_mt(
100  mt, partial, final_isomeric_state, initial_isomeric_state );
101  if( mt == search_mt )
102  {
103  for( size_t g = 0; g < flux.size(); ++g )
104  {
105  xs1 += xs->getValues()[g] * flux[g];
106  }
107  jDebugTaggedLine(
108  "found iza=" << iza << " mt=" << mt << " xs1=" << xs1 );
109  }
110  }
111  }
112  if( sum_flux == 0 ) return 0;
113  xs1 /= sum_flux;
114  return xs1;
115 }
