128 int Obiwan_diff::diff( const Library& liba,
129  const Library& libb,
130  std::ostream& out,
131  std::ostream& err )
132 {
133  bool have_difference = false;
134 
135  auto format = b_opts.get( "format", "y" );
136  int pos = b_opts.get( "pos", 1 );
137  float reltol = b_opts.get( "reltol", 0.01f );
138 
139  if( format == "eq" )
140  {
141  have_difference = !( liba == libb );
142  }
143  else if( format == "approx_eq" )
144  {
145  Insist( false, "approx_eq not implemented for library" );
146  }
147  else if( format == "y" )
148  {
149  Origen::SP_Diff d = diffLibrary( liba, libb, pos, reltol );
150  have_difference = d->have_difference();
151  out << d->to_string();
152  }
153  else if( format == "genneu" )
154  {
155  std::map<int, double>::const_iterator ita, itb, itc;
156 
157  std::map<int, double> amap = construct_neutron_yields_map( liba, pos );
158  std::map<int, double> bmap = construct_neutron_yields_map( libb, pos );
159  std::map<int, double> cmap = amap;
160  cmap.insert( bmap.begin(), bmap.end() );
161 
162  for( itc = cmap.begin(); itc != cmap.end(); ++itc )
163  {
164  int did_pid = itc->first;
165  ita = amap.find( did_pid );
166  itb = bmap.find( did_pid );
167  double coeffa = 0.0;
168  double coeffb = 0.0;
169  double relpct = 1e99;
170  std::string diffstr = "";
171  if( ita == amap.end() )
172  {
173  coeffb = itb->second;
174  if( coeffb != 0 )
175  {
176  diffstr = "DIFF(<)";
177  have_difference = true;
178  }
179  }
180  else if( itb == bmap.end() )
181  {
182  coeffa = ita->second;
183  if( coeffa != 0 )
184  {
185  diffstr = "DIFF(>)";
186  have_difference = true;
187  }
188  }
189  else
190  {
191  coeffa = ita->second;
192  coeffb = itb->second;
193  relpct =
194  100 *
195  std::fabs( ( 1e-99 + coeffb ) / ( 1e-99 + coeffa ) - 1.0 );
196  if( relpct > reltol )
197  {
198  diffstr = "DIFF(|)";
199  have_difference = true;
200  }
201  }
202  if( ita != amap.end() ) coeffa = ita->second;
203  if( itb != bmap.end() ) coeffb = itb->second;
204 
205  std::cout << did_pid << std::scientific << std::setw( 14 )
206  << std::setprecision( 5 ) << coeffa << " "
207  << std::scientific << std::setw( 14 )
208  << std::setprecision( 5 ) << coeffb << " "
209  << std::scientific << std::setw( 10 )
210  << std::setprecision( 1 ) << relpct << "%"
211  << " " << diffstr << std::endl;
212  }
213  }
214  else
215  {
216  b_opts.appendError(
217  "format=" + format +
218  " not recognized, must be (y|eq|approx_eq|genneu)" );
219  return -2;
220  }
221 
222  return have_difference ? 3 : 0;
223 }

306 int Obiwan_diff::execute( std::ostream& out, std::ostream& err )
307 {
308  // quick returns

316  return -1;

320  auto fileB = b_opts.get( "fileB", "" );
321 
322  int status;
323 
324  void* liba_ptr( loadMulti( fileA, b_opts ) );
325  std::string liba_type = b_opts.get<std::string>( "loadedDataType" );
326  void* libb_ptr( loadMulti( fileB, b_opts ) );
327  std::string libb_type = b_opts.get<std::string>( "loadedDataType" );
328  Insist( liba_type == libb_type, "Libraries are of differing types!" );
329 
330  if( liba_type == "Origen::Library" )
331  {
332  auto liba = Origen::SP_Library( static_cast<Library*>( liba_ptr ) );
333  auto libb = Origen::SP_Library( static_cast<Library*>( libb_ptr ) );
334  status = diff( *liba, *libb, out, err );
335  }
336  else if( liba_type == "Origen::StateSet" )
337  {
338  auto liba = Origen::SP_StateSet( static_cast<StateSet*>( liba_ptr ) );
339  auto libb = Origen::SP_StateSet( static_cast<StateSet*>( libb_ptr ) );
340  status = diff( *liba, *libb, out, err );
341  }
342  else
343  {
344  std::cerr << "Unknown library type!\n";

347
