264

267  ASSERT_TRUE( Origen::FakeFactory::Library_scale_pwr( lib ) );
268 
269  Origen::TransitionMatrixP::SP trx = lib.newsp_transition_matrix_at( 0 );
270 
271  // Set the transition matrix.
272  solver.set_transition_matrix( &*trx );
273 
274  // Turn the tallies on.
275  ScaleUtils::IO::DB opts;
276  opts.set<bool>( "tallyEnergyReleased", true );
277  opts.set<bool>( "tallyNumFissions", true );
278  opts.set<bool>( "tallyNumCaptures", true );
279  opts.set<bool>( "tallyAveConcentrations", true );
280  solver.set_opts( opts );
281 
282  // Step length & neutron flux.
283  double dt = 1 * 60 * 60 * 24;
284  double flux = 3.5e14;
285 
286  // Initial composition
287  Origen::Vec_Dbl n0( trx->get_itot(), 0.0 );
288  n0[trx->find_nuclide_guess( 922350 )] = 0.05;
289  n0[trx->find_nuclide_guess( 922380 )] = 0.95;
290 
291  // Solve
292  Origen::Vec_Dbl n( trx->get_itot(), 0.0 );
293  EXPECT_TRUE( solver.solve( n0, 3.5e14, dt, &n ) );
294 
295  // Get the tally results
296  const ScaleUtils::IO::DB& status = solver.status();
297  double energy_tally = status.get<double>( "energyReleased" );
298  double fission_tally = status.get<double>( "numFissions" );
299  double capture_tally = status.get<double>( "numCaptures" );
300 
301  // Get the total power at BOS and EOS
302  double BOS_power = trx->get_reaction_power( n0, 1, flux ) / 1e24;
303  double EOS_power = trx->get_reaction_power( n, 1, flux ) / 1e24;
304 
305  // Check that the energy tally is close. It should for such short step
306  double energy_ref = dt * ( BOS_power + EOS_power ) / 2;
307  EXPECT_NEAR( energy_ref, energy_tally, energy_ref * 1e-3 );
308 
309  // Check that the other tallies returned something
310  EXPECT_GT( fission_tally, 0.0 );
311  EXPECT_LT( fission_tally, 1.0 );
312  EXPECT_GT( capture_tally, 0.0 );
313  EXPECT_LT( capture_tally, 1.0 );
314 
315  // Check that the U238 Average concentration is reasnable
316  // This is a one day depletion
317  double u238 = status.get<double>( "ave_922380" );
318  EXPECT_LT( u238, 0.94999 );
319  EXPECT_GT( u238, 0.949 );
320 }
