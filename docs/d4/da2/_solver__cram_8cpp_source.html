348 bool Solver_cram::solve_impl( const Vec_Dbl& in,
349  double flux,
350  double dt,
351  Vec_Dbl* out )
352 {
353  jDebugLine( " CRAM solve time=" << dt << " flux=" << flux << " ..." );
354  Require( flux >= 0.0 );
355  Require( dt >= 0.0 );
356  Require( out != nullptr );
357  Require( ( this->check_opts( *d_opts, false ) ) );
358 
359  int energy_tally_idx = -1; // < 0 == not set
360  int fission_tally_idx = -1;
361  int capture_tally_idx = -1;
362  int concentration_tally_idx = -1;
363 
364  TransitionMatrixP trx_copy;
365  TransitionMatrixP* trx;
366 
367  Vec_Dbl n0( in );
368  Vec_Int tally_zaids; // used if average concentrations are tallied
369 
370  // Store the physical number of nuclides
371  int physical_itot = d_trx->get_itot();
372 
373  // Check if there are tallies, if yes, copy the TransitionMatrix
374  // NOTE: this copy could be avoided easily by giving the
375  // Transition matrix a "removeTallies" function. All you need to do
376  // is store the original sizes of the arrays and then truncate to those.
377 
378  if( d_opts->has<bool>( "tallyEnergyReleased" ) ||
379  d_opts->has<bool>( "tallyNumFissions" ) ||
380  d_opts->has<bool>( "tallyNumCaptures" ) ||
381  d_opts->has<bool>( "tallyAveConcentrations" ) )
382  {
383  trx_copy = TransitionMatrixP( *d_trx );
384  trx = &trx_copy;
385  }
386  else
387  {
388  trx = d_trx;
389  }
390 
391  // Add tallies (to the copy)
392  if( d_opts->has<bool>( "tallyEnergyReleased" ) &&
393  d_opts->get<bool>( "tallyEnergyReleased" ) )
394  {
395  energy_tally_idx = trx->add_energy_tally( flux );
396  n0.push_back( 0.0 );
397  }
398  if( d_opts->has<bool>( "tallyNumFissions" ) &&
399  d_opts->get<bool>( "tallyNumFissions" ) )
400  {
401  fission_tally_idx = trx->add_num_fissions_tally();
402  n0.push_back( 0.0 );
403  }
404  if( d_opts->has<bool>( "tallyNumCaptures" ) &&
405  d_opts->get<bool>( "tallyNumCaptures" ) )
406  {
407  capture_tally_idx = trx->add_num_captures_tally();
408  n0.push_back( 0.0 );
409  }
410  if( d_opts->has<bool>( "tallyAveConcentrations" ) &&
411  d_opts->get<bool>( "tallyAveConcentrations" ) )
412  {
413  concentration_tally_idx = trx->add_concentration_tallies( tally_zaids );
414  n0.insert( n0.end(), tally_zaids.size(), 0.0 );
415  }
416 
417  // Fold library.
418  DoubleList diag, a;
419  trx->fold_flux( flux, diag, a );
420 
421  // Get the number of nuclides.
422  int itot = trx->get_itot();
423 
424  // Ensure that the output array has enough space.
425  out->resize( n0.size(), 0.0 );
426 
427  // Check if there is a source term.
428  int source_order = -1;
429  if( d_src != nullptr )
430  for( int i = 0; i < itot; i++ )
431  if( d_src->at( i ) > 0.0 )
432  {
433  source_order = 0;
434  break;
435  }
436 
437  // Allocate and set the source term.
438  double** source_term = nullptr;
439  if( source_order > -1 )
440  {
441  source_term = (double**)std::malloc( itot * sizeof( double* ) );
442  for( int i = 0; i < itot; i++ )
443  source_term[i] =
444  (double*)std::calloc( source_order + 1, sizeof( double ) );
445 
446  for( int i = 0; i < d_src->size(); i++ )
447  source_term[i][0] = ( *d_src )[i];
448  }
449 
450  // Set removal rates.
451  if( d_lambda != nullptr )
452  for( int i = 0; i < physical_itot; i++ ) diag[i] -= ( *d_lambda )[i];
453 
454  // Is the flux zero? (We do not pass the flux itself to the solver.)
455  int zero_flux_step = 0;
456  if( flux == 0 ) zero_flux_step = 1;
457 
458  // Call the actual solver function (isn't this call beautiful).
459  int rv = kernel_cram( &( *out )[0],
460  &( n0[0] ),
461  source_term,
462  source_order,
463  dt,
464  itot,
465  trx->get_non(),
466  &( trx->get_non0_()->front() ),
467  &( trx->get_kd_()->front() ),
468  &diag[0],
469  &a[0],
470  &( trx->get_loc()->front() ),
471  zero_flux_step,
472  d_opts->get<bool>( "isAdjoint" ),
473  d_opts->get<int>( "cramOrder" ),
474  d_opts->get<int>( "cramInternalSubsteps" ),
475  d_opts->get<int>( "cramRemoveNegatives" ),
476  d_opts->get<double>( "cramResultsCutoff" ),
477  &( trx->get_nucl()->front() ) );
478 
479  // Free memory.
480  if( source_order > -1 )
481  {
482  for( int i = 0; i < itot; ++i ) std::free( source_term[i] );
483  std::free( source_term );
484  }
485 
486  // Add status flag to the internal status container.
487  d_opts->set<int>( "statusFlag", rv );
488  if( rv != 0 )
489  {
490  std::stringstream ss;
491  ss << "CRAM solution failed with exit code " << rv << std::endl;
492  d_opts->appendError( ss.str() );
493  }
494 
495  // Extract the tally results
496  if( energy_tally_idx > 0 )
497  d_opts->set<double>( "energyReleased", ( *out )[energy_tally_idx] );
498 
499  if( fission_tally_idx > 0 )
500  d_opts->set<double>( "numFissions", ( *out )[fission_tally_idx] );
501 
502  if( capture_tally_idx > 0 )
503  d_opts->set<double>( "numCaptures", ( *out )[capture_tally_idx] );
504 
505  if( concentration_tally_idx > 0 )
506  for( int i = 0; i < tally_zaids.size(); ++i )
507  {
508  char param_name[512];
509  sprintf( param_name, "ave_%d", tally_zaids[i] );
510  double val = ( *out )[concentration_tally_idx + i] / dt;
511 
512  d_opts->set<double>( param_name, val );
513  }
514 
515  out->resize( in.size() );
516 
517  return ( rv == 0 );
518 }
