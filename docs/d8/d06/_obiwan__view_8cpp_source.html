320 int view( const Origen::Library& lib,
321  ScaleUtils::IO::DB opts,
322  std::ostream& out,
323  std::ostream& err )
324 {
325  // format
326  std::string format = opts.get( "format", "coeff" );
327  std::string idform = opts.get<std::string>( "idform", "" );
328  bool output_zeros = opts.get( "output_zeros", true );
329 
330  // output burnups
331  if( format == "burnups" )
332  {
333  bool transpose = opts.get( "transpose", true );
334  int prec = opts.get( "prec", 4 );
335  printBurnups( lib, transpose, prec, out );
336  }
337  // output is JSON format
338  else if( format == "json" )
339  {
340  out << lib.to_json().toStyledString();
341  }
342  // text output of coefficients
343  else if( format == "" || format == "coeff" || format == "coefft" )
344  {
345  int sizzzaaa = opts.get( "sizzzaaa", 0 );
346  int tid = opts.get( "tid", 0 );
347  bool transpose = opts.get( "transpose", true );
348  double flux = opts.get( "flux", 1.0 );
349  bool coefft = format == "coefft" ? true : false;
350  int prec = opts.get( "prec", 4 );
351  int pos = opts.get( "pos", 1 );
352  printTransitionCoeff( lib,
353  sizzzaaa,
354  tid,
355  transpose,
356  output_zeros,
357  flux,
358  coefft,
359  idform,
360  prec,
361  pos,
362  out );
363  }
364  // tags
365  else if( format == "tags" )
366  {
367  std::string path = opts.get<std::string>( "path" );
368  printTags_json( lib, path );
369  }
370  // reaction
371  else if( format == "reaction" )
372  {
373  printReactionTransitions( lib, output_zeros, idform, out );
374  }
375  // text output of nuclide
376  else if( format == "nucl" )
377  {
378  int sizzzaaa = opts.get( "sizzzaaa", 0 );
379  bool transpose = opts.get( "transpose", true );
380  int prec = opts.get( "prec", 4 );
381  printNuclideInfo( lib, sizzzaaa, transpose, idform, prec, out );
382  }
383  // print data sources
384  else if( format == "genneu" || format == "nprodr" || format == "fisxs" ||
385  format == "loxs" )
386  {
387  int sizzzaaa = opts.get( "sizzzaaa", 0 );
388  bool transpose = opts.get( "transpose", true );
389  int prec = opts.get( "prec", 4 );
390  int pos = opts.get( "pos", 1 );
391  if( format == "fisxs" )
392  {
393  printFissionXs( lib, sizzzaaa, transpose, idform, prec, pos, out );
394  }
395  else if( format == "loxs" )
396  {
397  printLossXs( lib, sizzzaaa, transpose, idform, prec, pos, out );
398  }
399  else if( format == "genneu" )
400  {
401  printNeutronYields( lib, pos, out );
402  }
403  else if( format == "nprodr" )
404  {
405  printNeutronProductionRatio( lib, pos, out );
406  }
407  }
408  // output transition system channel
409  else if( format == "ts" )
410  {
411  opts.appendError(
412  "TransitionSystem (ts) output is currently disabled" );
413  return -3;
414  }
415  // output graph
416  else if( format == "graphviz" )
417  {
418  Origen::TransitionGroupFilter tfilter;
419  Origen::NuclideGroupFilter nfilter;
420  COPY_LIST( opts, IntegerList, nfilter, allow_s );
421  COPY_LIST( opts, IntegerList, nfilter, allow_i );
422  COPY_LIST( opts, IntegerList, nfilter, allow_z );
423  COPY_LIST( opts, IntegerList, nfilter, allow_a );
424  COPY_LIST( opts, IntegerList, tfilter, allow_transition_id );
425  tfilter.allow_decay = opts.get( "allow_decay", true );
426  tfilter.allow_reaction = opts.get( "allow_reaction", true );
427  tfilter.allow_fission = opts.get( "allow_fission", false );
428  bool include_byproducts = opts.get( "include_byproducts", false );
429  int pos = opts.get( "pos", 1 );
430  // std::vector<double> nd = ndtest();
431  // std::ofstream ofs("o.txt");
432  // graphvizLibrary(*lib,pos,&nd,nfilter,tfilter);
433  graphvizLibrary(
434  lib, pos, nullptr, nfilter, tfilter, include_byproducts, out );
435  }
436  // error
437  else
438  {
439  opts.appendError( "view format={" + format +
440  "} unknown for Origen::Library type (see help)" );
441  return -4;
442  }
443  return 0;
