341  H-1 (n,g) // A reaction that already exists
342  H-1 (n,2n) // A reaction that does not exist (yes its silly)

345  B-11 (n,na) // A reaction which (at the time of writing this) is

347  U-235 (n,f) // This should not break fission.
348  */
349 
350  std::vector<int> user_nuclide_list = {1001, 1002, 5011, 92235};
351  std::vector<int> user_num_mts = {2, 0, 2, 1};

383  EXPECT_TRUE( xs[2] != 0.0 );
384  EXPECT_TRUE( xs[3] == 0.0 );
385  EXPECT_TRUE( xs[2] != 0.0 );
386 
387  // Set the cross-sections.
388  for( int i = 0; i < xs.size(); ++i ) xs[i] = ( i + 1 ) * 1000;
389 
390  tmu.update_user_xs(

392

394  for( int i = 0; i < xs.size(); ++i ) xs[i] = 0;
395

405

408  diag,
409  offdiag ); // The 1e24 flux causes the elements

411
