141 ScaleUtils::IO::DB baseOptions( Standard::CommandLine& cmd )
142 {
143  ScaleUtils::IO::DB base;
144 
145  // overwrite
146  base.set( "overwrite", cmd.exists( "-i" ) ? true : false );
147 
148  // verbose
149  base.set( "verbose", cmd.exists( "-verbose" ) ? true : false );
150 
151  if( cmd.getValue( "-format", "" ) != "" &&
152  cmd.getValue( "-version" ) != "" )
153  {
154  base.appendError( "cannot specify both -format and -version" );
155  }
156 
157  // default for version (always overridden by format if non-null)
158  std::string version = cmd.getValue( "-version", "6.2" );
159  base.set( "desiredTargetScaleVersion", version );
160 
161  // remove parts
162  base.set( "removeMass", cmd.exists( "-remove_mass" ) );
163  base.set( "removeAbundance", cmd.exists( "-remove_abund" ) );
164 
165  // desired format
166  base.set( "desiredFileFormat", cmd.getValue( "-format", "" ) );
167 
168  // get data type to read
169  populateObiwan_dataType( &base, cmd.getValue( "-type", "" ) );
170 
171  // default for targetEndian
172  std::string endian = ( version == "6.1" ) ? "big" : "native";
173  base.set( "targetEndian", cmd.getValue( "-endian", endian ) );
174 
175  base.set( "targetDirectory", cmd.getValue( "-dir", "." ) );
176  base.set( "thinningEnabled", cmd.getValueAs( "-thin", false ) );
177  if( base.get<bool>( "thinningEnabled" ) )
178  {
179  base.setFromJson<DoubleList*>( "timeValues",
180  cmd.getValue( "-tvals", "[]" ) );
181  base.set( "interpMethod", cmd.getValue( "-interp", "near" ) );
182  // base.set("deltaTimeMin",cmd.getValueAs("-dtmin",100.));
183  // base.set("deltaTimeMax",cmd.getValueAs("-dtmax",5000.));
184  base.set( "responseVariable", cmd.getValue( "-rvar", "loxs" ) );
185  base.set( "responseNuclide", cmd.getValue( "-rnuc", "pu240" ) );
186  base.set( "verbose", true );
187  // base.set("responseTolerance",cmd.getValueAs("-rtol",1e-3));
188  }
189  if( cmd.exists( "-setbu" ) )
190  {
191  base.setFromJson<DoubleList*>( "newBurnups",
192  cmd.getValue( "-setbu", "[]" ) );
193  }
194  return base;
195 }

234 int Obiwan_convert::execute( std::ostream& out, std::ostream& err )
235 {
236  // quick returns
237  if( b_opts.get( "help", false ) )
238  {
239  out << help() << std::endl;
240  return 1;
241  }
242  else if( b_opts.numError() > 0 )
243  {
244  out << b_opts.dumpErrorStack();
245  return -1;
246  }
247 
248  auto files = b_opts.get<StringList*>( "paths" );
249  ScaleUtils::IO::Table2D table;
250  table.setTitle( "originalFile" );
251  table.addX( "dataType" );
252  table.addX( "loadedFormat" );
253  table.addX( "convertedFile" );
254  table.addX( "savedFormat" );
255  table.addX( "runtime(ms)" );
256  table.addX( "errors" );
257 
258  // Loop over files.
259  for( const auto& path : *files )
260  {
261  // variables
262  SP_Library lib;
263  SP_StateSet mat;
264  ScaleUtils::IO::DB opts = b_opts.clone();
265  bool verbose = opts.get<bool>( "verbose", false );
266 
267  // load data
268  void* obj_ptr = loadMulti( path, opts );
269 
270  // perform conversion
271  std::string loadedDataType = opts.get( "loadedDataType", "UNKNOWN" );
272 
273  if( opts.numError() == 0 )
274  {
275  // update what to write
276  opts.set( "fileFormat", opts.get( "desiredFileFormat", "" ) );
277  opts.set( "targetScaleVersion",
278  opts.get( "desiredTargetScaleVersion", "" ) );
279 
280  if( loadedDataType == "Origen::Library" )
281  {
282  lib = Origen::SP_Library( (Library*)( obj_ptr ) );
283  Check( lib );
284  convertLibrary( *lib, path, opts );
285  }
286  else if( loadedDataType == "Origen::StateSet" )
287  {
288  mat = Origen::SP_StateSet( (StateSet*)( obj_ptr ) );
289  Check( mat );
290  convertStateSet( *mat, path, opts );
291  }
292  }
293 
294  // new row of table
295  table.addRow();
296  table.addY( ScaleUtils::IO::getBaseName( path ) );
297  table.addCell( loadedDataType );
298  table.addCell( opts.get( "loadedFileFormat", "-" ) );
299  table.addCell( opts.numError() == 0 ? opts.get( "fileName", "-" )
300  : "-" );
301  table.addCell( opts.get( "savedFileFormat", "-" ) );
302 
303  // output options in JSON
304  if( verbose ) err << opts << std::endl;
305 
306  // add runtime
307  table.addCell( nprintf( " %.0e",
308  opts.get<double>( "loadTime", 0.0 ) +
309  opts.get<double>( "saveTime", 0.0 ) ) );
310 
311  // add errors
312  if( opts.numError() == 1 )
313  {
314  table.addCell( opts.errorStack()[0] );
315  }
316  else if( opts.numError() > 1 )
317  {
318  std::string file =
319  opts.get( "fileName", "obiwan-convert" ) + ".err";
320  table.addCell( "see file='" + file + "'" );
321  std::ofstream ofs( file );
322  ofs << opts.dumpErrorStack();
323  }
324  else
325  {
326  table.addCell( "-" );
327  }
328 
329  // clear everything so we don't get a throw
330  opts.clear();
331  }
332  out << table.straight().text( 12, 8 );
333 
334  return 0;
335 }
