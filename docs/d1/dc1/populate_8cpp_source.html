34 //! idform can include various symbols to be substituted
35 // $S - sublib integer
36 // $I - isomeric state integer
37 // $ZZZ - up to 3 digits of atomic number (also Z, ZZ)
38 // $AAA - up to 3 digits of mass number (also A, AA)
39 // $EE - up to 2 characters elemental symbol (also Ee,ee for different
40 // case)
41 // $M - metastable indicator 'M' for first, 'M2' for second, etc. (also
42 // m)
43 // $TY - upper case sublib name
44 // $ty - lower case sublib name
45 std::string stringifyNuclide( int id, std::string idform )
46 {
47  int s, i, z, a;
48  ScaleData::Utils::split_pizzzaaa( id, s, i, z, a );
49 
50  // quick return on standard SIZZZAAA form
51  if( idform == "$S$I$ZZZ$AAA" || idform == "" )
52  {
53  return nprintf( "%07d", id );
54  }
55 
56  std::string str = idform;
57 
58  // get various element representations
59  std::string ee = stringToLower( ScaleData::Utils::z_to_symbol( z ) );
60  std::string EE = stringToUpper( ee );
61  std::string Ee = ee;
62  std::transform( Ee.begin(), Ee.begin() + 1, Ee.begin(), ::toupper );
63  std::string TY = stringifySublib( s ).c_str();
64  std::string ty = stringToLower( TY );
65 
66  // get various metastable representations
67  std::string M = "";
68  if( i > 0 ) M = "M";
69  if( i > 1 ) M += std::to_string( i );
70  std::string m = stringToLower( M );
71 
72  str = findAndReplace( str, "$ZZZ", nprintf( "%03d", z ) );
73  str = findAndReplace( str, "$ZZ", nprintf( "%02d", z ) );
74  str = findAndReplace( str, "$Z", nprintf( "%d", z ) );
75  str = findAndReplace( str, "$AAA", nprintf( "%03d", a ) );
76  str = findAndReplace( str, "$AA", nprintf( "%02d", a ) );
77  str = findAndReplace( str, "$A", nprintf( "%d", a ) );
78  str = findAndReplace( str, "$S", nprintf( "%d", s ) );
79  str = findAndReplace( str, "$I", nprintf( "%d", i ) );
80  str = findAndReplace( str, "$EE", nprintf( "%s", EE.c_str() ) );
81  str = findAndReplace( str, "$ee", nprintf( "%s", ee.c_str() ) );
82  str = findAndReplace( str, "$Ee", nprintf( "%s", Ee.c_str() ) );
83  str = findAndReplace( str, "$M", nprintf( "%s", M.c_str() ) );
84  str = findAndReplace( str, "$m", nprintf( "%s", m.c_str() ) );

598

600  ScaleData::Utils::transform_mt(
601  mt, partial, final_isomeric_state, initial_isomeric_state );
602 
603  // filter MTs user doesn't want
604  const auto& allowed_mts = rr->allowed_mts();
605  auto ita = allowed_mts.find( mt );
606  if( ita == allowed_mts.end() )
607  {
608  rr->set_disallowed_mts( Vec_Int( {mt} ), true );
609  jDebugTaggedLine(
610  "import: establishing on-fly rule to disallow izzzaaa="
611  << izzzaaa
612  << " mt="
613  << mt );
614  continue;
615  }
616  else if( !ita->second )
617  {
618  jDebugTaggedLine(
619  "import: disallowing izzzaaa=" << izzzaaa << " mt=" << mt
620  << " by established rule" );
621  continue;
622  }
623 
624  // filter isomeric states we can't handle for reaction
625  if( final_isomeric_state > rr->max_isomeric_state() )
626  {
627  jDebugTaggedLine( "import: disallowing final_isomeric_state="
628  << final_isomeric_state
629  << " for mt="
630  << mt
631  << " because greater than max allowed"
632  " of "
633  << rr->max_isomeric_state() );
634  continue;
635  }
636 
637  // if we get here, this data will be imported
638  jDebugTaggedLine(
639  "import: importing izzzaaa=" << izzzaaa << " mt=" << mt
640  << " initial_isomeric_state="
641  << initial_isomeric_state
642  << " final_isomeric_state="
643  << final_isomeric_state );
644 
645  // partial cross sections should satisfy
646  Check( partial ? i == initial_isomeric_state : true );
647 
648  // create a new xs data entry
649  ReactionResource::Nuclide::Xs& a_xd = a_nd[mt];
650 
651  // save partial/isomeric cross section
652  // note that the total is not given if partials are so it's
653  // important that we don't normalize in order to sum to
654  // get the total channel cross section (totxs)
655  if( partial )
656  {
657  // remember the type of this data
658  a_xd.yield_type = YIELD_ISOMERIC;
659 
660  jDebugTaggedLine( "import: nuclide has a partial xs" );
661 
662  // resize to fit this data
663  a_xd.pyield.resize( 1 + final_isomeric_state );
664  Vec_Flt& mxs = a_xd.pyield[final_isomeric_state];
665 
666  // store partial yields
667  mxs.resize( mg_size );
668  for( int g = 0; g < mg_size; ++g )
669  {
670  mxs[g] = xs->getValues()[g];
671  // would like to not have to have this check!
672  if( mxs[g] < 0.0 )
673  {
674  jDebugBlock(
675  jDebugTaggedWarning( "WARNING: nuclide="
676  << izzzaaa
677  << " has negative mxs["
678  << g
679  << "]="
680  << mxs[g]
681  << " for final_isomeric_state="
682  << final_isomeric_state ); );
683  mxs[g] = 0.0;
684  }
685  Check( mxs[g] >= 0.0 );
686  }
687  rr->map_xs( &mxs ); // map to library group
688 
689  continue;
690  }
691 
692  // must be a channel total
693  jDebugTaggedLine(
694  "import: fallback to treating as total channel xs" );
695 
696  // Only overwrite type if unknown to allow for a DISTRIBUTED
697  // yield which also uses totxs, e.g. fission
698  Check( a_xd.yield_type == YIELD_UNKNOWN );
699  if( mt == 18 )
700  {
701  a_xd.yield_type = YIELD_DISTRIBUTED;
702 
703  // Add a nubar because we always want to have a yield
704  // of neutrons if we have fission
705  bool using_approx_nubar = false;
706  if( yield_map.count( 18 ) == 0 )
707  {
708  using_approx_nubar = true;
709  // Use Journal of Nuclear and Radiochemical Sciences, Vol.
710  // 9, No.1, pp. 19-25, 2008
711  float nubar = 0.27318 * z - 22.7734;
712  if( nubar < 0 ) nubar = 0;
713  auto nu = new CrossSection1d();
714  nu->setSize( mg_size );
715  for( int g = 0; g < mg_size; ++g )
716  nu->getValues()[g] = nubar;
717  yield_map[18] = nu;
718  }
719 
720  // Associate nu data (fission will be done below as part of
721  // totxs.)
722  auto& a_xdd = a_nd[18];
723  auto& direct_yield = a_xdd.dyield[1]; // 1=NEUTRON
724  direct_yield.resize( mg_size );
725  auto nu = yield_map[18];
726  Check( nu != nullptr );
727  for( int g = 0; g < mg_size; ++g )
728  {
729  direct_yield[g] = nu->getValues()[g] * xs->getValues()[g];
730  }
731 
732  if( using_approx_nubar ) delete yield_map[18]; // created above
733 
734  rr->map_xs( &direct_yield ); // map to library group
735  }
736  else
737  {
738  a_xd.yield_type = YIELD_SIMPLE;
739  }
740 
741  // store channel total xs (as opposed to partial)
742  Vec_Flt& totxs = a_xd.totxs;
743  totxs.resize( mg_size );
744  for( int g = 0; g < mg_size; ++g )
745  {
746  totxs[g] = xs->getValues()[g];
747  // would like to not have to have this check!
748  if( totxs[g] < 0.0 )
749  {
750  jDebugBlock( jDebugTaggedWarning( "WARNING: nuclide="
751  << izzzaaa
752  << " has negative totxs["
753  << g
754  << "]="
755  << totxs[g] ); );
756  totxs[g] = 0.0;
757  }
758  Check( totxs[g] >= 0.0 );
759  }
760  rr->map_xs( &totxs ); // map to library group
761  }
762  }
