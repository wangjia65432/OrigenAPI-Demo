101 std::string expandPaths( const std::string& sourcePath,
102  std::vector<int>& coeffPosList,
103  std::string& coeffDataKey )
104 {
105  // start to parse data path
106  auto components = ScaleUtils::IO::splitString( "/", sourcePath );
107  if( components.size() != 2 )
108  {
109  return "The -data=" + sourcePath +
110  " argument must contain two '/'-separated components!";
111  }
112 
113  // extract any position specifier, split coeff[*]-->coeff,*
114  // coeff[1]-->coeff,1
115  // coeff[1|2]-->coeff,1|2
116  std::string pos = "*";
117  auto posPath = ScaleUtils::IO::splitString( "[", components[0] );
118  if( posPath.size() > 1 )
119  {
120  components[0] = posPath[0];
121  pos = posPath[1];
122  // remove trailing ]
123  pos = ScaleUtils::IO::findAndReplace( pos, "]", "" );
124  }
125 
126  // check coeff
127  if( components[0] != "coeff" )
128  {
129  return "The -data=" + sourcePath +
130  " argument must contain the first component of 'coeff'!";
131  }
132 
133  // check data key
134  coeffDataKey = components[1];
135  if( coeffDataKey != "n_production_xs" && coeffDataKey !="nu" )
136  {
137  return "The -data=" + sourcePath +
138  " is not recognized. Only coeff[POS]/X for "
139  "X=n_production_xs,nu is supported!";
140  }
141 
142  // determine list of positions
143  coeffPosList.clear();
144  if( pos != "*" )
145  {
146  auto list = ScaleUtils::IO::splitString( "|", pos );
147  for( auto l : list )
148  {
149  coeffPosList.push_back( ScaleUtils::IO::fromString<int>( l, -1 ) );
150  if( coeffPosList.back() <= 0 )
151  {
152  return "The -data=" + sourcePath + " list argument=" + l +
153  " must be a valid integer and >0";
154  }
155  }
156  }
157  return "";
158 }

243 std::string applyPatch( SP_Library fromLib,
244  IntegerList* coeffPosList,
245  const std::string& coeffData,
246  SP_Library lib,
247  int pos )
248 {
249  // need to fill this many values
250  size_t nSink = pos == 0 ? lib->transition_coeff_size() : 1;
251  size_t nSource = fromLib == nullptr ? 1 : fromLib->transition_coeff_size();
252 
253  // single position for the user-defined patch so we just copy
254  if( coeffPosList->size() == 1 )
255  {
256  auto single = coeffPosList->at( 0 );
257  for( size_t i = 1; i < nSink; ++i ) coeffPosList->push_back( single );
258  }
259  // coeffPosList is empty and we need to create it
260  else if( coeffPosList->size() == 0 )
261  {
262  // the source only has one position so we just copy
263  if( nSource == 1 )
264  {
265  for( size_t i = 0; i < nSink; ++i ) coeffPosList->push_back( 1 );
266  }
267  // otherwise there should be 1-to-1 mapping
268  else
269  {
270  for( size_t i = 0; i < nSource; ++i )
271  coeffPosList->push_back( i + 1 );
272  }
273  }
274 
275  // check correct sizes
276  if( coeffPosList->size() != nSink )
277  {
278  return "the number of positions to extract for the patch (" +
279  std::to_string( coeffPosList->size() ) +
280  ") must match the number of positions on the target (" +
281  std::to_string( nSink ) + ").";
282  }
283 
284  // check coeffPosList positions are valid
285  if( fromLib != nullptr )
286  {
287  for( auto j : *coeffPosList )
288  {
289  if( j < 1 || j > nSource )
290  return "the position on the -from library (" +
291  std::to_string( j ) + ") is not valid (>=1 and <=" +
292  std::to_string( nSource ) + ")";
293  }
294  }
295 
296  // start the patching
297  for( size_t i = 0; i < nSink; ++i )
298  {
299  // patch this position on sink
300  if( pos == 0 || pos == i + 1 )
301  {
302  // get copies
303  auto tc = lib->transition_coeff_at( i );
304  auto npxs = tc.neutron_yields();
305  // fix nans
306  for(auto& xs : npxs )
307  {
308  if( std::isnan(xs) )xs=0;
309  }
310  size_t n = npxs.size();
311 
312  // replace with zeros if source was NULL
313  if( fromLib == nullptr )
314  {
315  npxs.clear();
316  npxs.resize( n, 0.0 );
317  }
318  // otherwise actually query source
319  else
320  {
321  // get index on the source
322  int fromPos =
323  pos > 0 ? coeffPosList->at( 0 ) : coeffPosList->at( i );
324  auto& fromTc = fromLib->transition_coeff_at( fromPos - 1 );
325  size_t fromN = fromTc.neutron_yields_size();
326  if( n != fromN )
327  return "sizes of the n_production_xs are not the same on "
328  "the -from library (" +
329  std::to_string( fromN ) + ") and the destination (" +
330  std::to_string( n ) + ")!";
331  //just replace neutron production xs
332  if( coeffData=="n_production_xs" )
333  {
334  npxs = fromTc.neutron_yields();
335  }
336  //only replace nu and hack it as n_production_xs/sigma_f
337  else if( coeffData=="nu" )
338  {
339  auto new_npxs = fromTc.neutron_yields();
340  // fix nans
341  for(auto& xs : new_npxs )
342  {
343  if( std::isnan(xs) )xs=0;
344  }
345  auto new_sigf = fromTc.fission_xs();
346  auto old_sigf = tc.fission_xs();
347  size_t sigf_offset = new_npxs.size()-new_sigf.size();
348  for(size_t i=sigf_offset; i<npxs.size(); ++i)
349  {
350  if( new_sigf[i-sigf_offset]>0.0 )
351  {
352  auto nu = 1+new_npxs[i]/new_sigf[i-sigf_offset];
353  npxs[i] = nu*old_sigf[i-sigf_offset];
354  }
355  }
356  }
357  }
358  tc.set_neutron_yields( npxs );
359  lib->set_transition_coeff_at( tc, i );
360  }
361  }
362  return "";

364 int Obiwan_patch::execute( std::ostream& out, std::ostream& err )
365 {
366  // quick returns
367  if( b_opts.get( "help", false ) )
368  {
369  out << help() << std::endl;
370  return 1;
371  }
372  else if( b_opts.numError() > 0 )
373  {
374  out << b_opts.dumpErrorStack();
375  return -1;
376  }
377 
378  // data to generate patch from
379  auto fromPath = b_opts.get( "fromPath", "" );
380  SP_Library fromLib;
381 
382  // data to insert patch into
383  auto path = b_opts.get( "path", "" );
384  SP_Library lib;
385 
386  // variables
387  ScaleUtils::IO::DB opts = b_opts.clone();
388  bool verbose = opts.get<bool>( "verbose", false );
389 
390  // load data
391  void* obj_ptr = loadMulti( path, opts );
392  std::string loadedDataType = opts.get( "loadedDataType", "UNKNOWN" );
393  if( opts.numError() != 0 )
394  {
395  opts.dumpErrorStack();
396  return -1;
397  }
398  if( loadedDataType == "Origen::Library" )
399  {
400  lib = SP_Library( (Library*)( obj_ptr ) );
401  if( fromPath != "NULL" )
402  fromLib = SP_Library( loadLibrary( fromPath, opts ) );
403  }
404  if( opts.numError() != 0 )
405  {
406  opts.dumpErrorStack();
407  return -1;
408  }
409 
410  // extract all positions of data on fromLib
411  // check that if multiple positions was specified for fromLib, then lib
412  // has same number of positions
413  // if a single position of data was specified, copy it to all positions
414  auto error = applyPatch( fromLib,
415  opts.get<IntegerList*>( "coeffPosList" ),
416  opts.get<std::string>( "coeffDataKey" ),
417  lib,
418  opts.get<int>( "pos" ) );
419  if( error != "" ) opts.appendError( error );
420 
421  // output options in JSON
422  opts.set( "overwrite", true );
423  saveLibrary( *lib, path, opts );
424  if( verbose ) err << opts << std::endl;
425 
426  // final chance for error
427  if( opts.numError() != 0 )
428  {
429  opts.dumpErrorStack();
430  return -1;
431  }
432 
433  return 0;
434 }
