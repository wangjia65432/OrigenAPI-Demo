277  * \brief This function should check the provided tag manager against the
278  * current
279  * tag manager to ensure a perfect match of tag names. This is achieved
280  * by calling the internal compare function twice, switching the
281  * arguments. If the internal compare function returns a null set both
282  * times, the two match. Otherwise, they don't match.
283  *
284  * \param other Another TagManager to compare against the current TagManager.
285  * \return bool - True if all of the tag names on the provided TagManager
286  * perfectly match those on the current TagManager.
287  * False otherwise.

289 bool TagManager::match( const TagManager& other ) const
290 {
291  Vec_Str tagsPresent;
292  Vec_Str holder;
293  Vec_Str otherTags;
294  Vec_Str::const_iterator hit;
295  tagsPresent = this->listIdTags();
296  holder = this->listInterpTags();
297  for( hit = holder.begin(); hit != holder.end(); ++hit )
298  {
299  tagsPresent.push_back( *hit );
300  }
301  holder.clear();
302  otherTags = other.listIdTags();
303  holder = other.listInterpTags();
304  for( hit = holder.begin(); hit != holder.end(); ++hit )
305  {
306  otherTags.push_back( *hit );
307  }
308  if( !internalCompare( tagsPresent, other ).empty() ) return false;
309  if( !internalCompare( otherTags, *this ).empty() ) return false;
310  return true;

370  * \brief Checks for missing tags on a Library's TagManager object,
371  * based on a set of desired tag names.
372  *
373  * This private function accepts a vector of strings that are desired
374  * tag names and a full TagManager instance from a Library. The names
375  * of the tags are taken from the full TagManager instance into a vector
376  * of strings that are the tag names present. The list of the desired
377  * tags is iterated through and if no match is found in the tags present,
378  * it is added to the list of missing tags. The list of missing tags is
379  * then returned.
380  *
381  * \param desired List of desired tag names that we are hoping to find in the
382  * other library.
383  * \param other TagManager from the library we are checking for the desired
384  * tags.
385  * \return List of tag names that were desired but not found on the library
386  * in question.
387  */
388 Vec_Str TagManager::internalCompare( Vec_Str desired,
389  const TagManager& other ) const
390 {
391  Vec_Str present;
392  Vec_Str missing;
393  Vec_Str holder;
394  Vec_Str::const_iterator pit;
395  Vec_Str::const_iterator dit;
396  Vec_Str::const_iterator hit;
397 
398  present = other.listIdTags();
399  holder = other.listInterpTags();
400  for( hit = holder.begin(); hit != holder.end(); ++hit )
401  {
402  present.push_back( *hit );
403  }
404  for( dit = desired.begin(); dit != desired.end(); ++dit )
405  {
406  bool found = false;
407  for( pit = present.begin(); pit != present.end(); ++pit )
408  {
409  if( ( *dit ).compare( *pit ) == 0 )
410  {
411  found = true;
412  break;
413  }
414  }
415  if( !found ) missing.push_back( *dit );
416  }
417  return ( missing );
418 } // internalCompare

456 /*!
457  * \brief Function to check the tag managers from a set of TagManagers
458  * (e.g., from a set of Library objects) to enssure that they have all
459  * of the tags requested in a reference TagManager. Then checks to see
460  * if the libraries have any other interpolable tags. If so, these are
461  * collected along with all of the values for that tag on all of the
462  * TagManagers and returned.
463  *
464  * \param tagManList List of all of the tag managers in question.
465  *
466  * \param tm Reference TagManager.
467  *
468  * \return List of implicitTag structures, which contains the name and a
469  * vector of values corresponding to that tag name from all of
470  * the libraries.
471  */
472 std::vector<implicitTag> checkImplicitTags(
473  const std::vector<TagManager>& tagManList, const TagManager& tm )
474 {
475  std::vector<implicitTag> tagSpans;
476  for( auto& tagMan : tagManList )
477  {
478  Insist(
479  tm.compare( tagMan ).size() == 0,
480  "A Library does not contain necessary tags for interpolation." );
481  for( auto& tag : tagMan.listInterpTags() )
482  {
483  if( !tm.hasTag( tag ) )
484  {
485  bool tmpbool = true;
486  while( tmpbool )
487  {
488  for( auto& implied : tagSpans )
489  {
490  if( implied.name == tag )
491  {
492  implied.values.push_back(
493  tagMan.getInterpTag( tag ) );
494  tmpbool = false;
495  break;
496  }
497  }
498  if( !tmpbool ) break;
499  implicitTag newTag;
500  newTag.name = tag;
501  newTag.values.push_back( tagMan.getInterpTag( tag ) );
502  tagSpans.push_back( newTag );
503  tmpbool = false;
504  }
505  }
506  }
507  }
508  return ( tagSpans );
509 }

540 /// parse the command line argument that has tags of the form
541 /// tag1,tag2=,tag3=val3,... convenient for use with TagManager
542 std::string parseTags( const std::string& str,
543  Vec_Str* tags,
544  Vec_Str* values,
545  Vec_Bool* modify )
546 {
547  Vec_Str tmptags = ScaleUtils::IO::splitString( ",", str );
548  modify->resize( 0 );
549  values->resize( 0 );
550  tags->resize( 0 );
551 
552  for( auto tag : tmptags )
553  {
554  modify->push_back( false );
555  values->push_back( "" );
556  tags->push_back( "" );
557  Vec_Str left_right = ScaleUtils::IO::splitString( "=", tag );
558  if( left_right.size() == 2 )
559  {
560  modify->back() = true;
561  if( left_right[1] != "" )
562  {
563  values->back() = left_right[1];
564  tags->back() = left_right[0];
565  }
566  }
567  else if( left_right.size() == 1 && left_right[0] != "" )
568  {
569  modify->back() = true;
570  tags->back() = left_right[0];
571  }
572  else if( left_right.size() > 2 )
573  {
574  return "tag expression: " + tag + " has too many =";
575  }
576  }
577  Ensure( modify->size() == tags->size() );
578  Ensure( values->size() == tags->size() );
579  return "";
580 }
