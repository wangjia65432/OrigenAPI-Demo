100 {
101  if( states_size() <= 1 ) return true;

103  {
104  const State& ci = states_at( i );
105  const State& cp = states_at( i - 1 );
106  if( cp.num_neutron_groups() != ci.num_neutron_groups() ) return false;

344 //! \brief Create a set of interpolated responses from a viewer list
345 template <typename T>
346 typename T::GD_t::Vec_Rptr StateSet::create_interp_responses(
347  const std::vector<T>& viewer_list ) const
348 {
349  size_t nx = viewer_list.size();
350  size_t nr = ( nx == 0 ) ? 0 : viewer_list[0].nr();
351 
352  typename T::GD_t::Vec_Rptr response( nr * nx );
353 
354  size_t f = 0;
355  for( auto& v : viewer_list )
356  {
357  // get response grid
358  const typename T::GD_t::Vec_Rptr& r = v.r();
359  Check( v.nr() == nr );
360  for( size_t j = 0; j < nr; ++j )
361  {
362  response[f] = r[j];
363  ++f;
364  }
365  }
366  Ensure( f == nr * nx );
367  return response;
368 }

424 /**
425  * \brief Generate an interpolation view of the StateSet by mapping a set of
426  * States
427  * onto GridData;
428  *
429  * \remark Grid extents defined by times (interpolated on time)
430  */
431 template <typename View>
432 typename View::Type::SP StateSet::interpolate_view(
433  float time,
434  const std::vector<float>& times,
435  const std::vector<View>& viewer_list,
436  const View& prototype ) const
437 {
438  // get responses
439  auto&& response = create_interp_responses( viewer_list );
440 
441  size_t nx = times.size();
442 
443  // Grid b_state object
444  typename View::GD_t gd;
445 
446  // set x-grid (1D with times in this case)
447  typename View::GD_t::Vec_I gextents( {nx} ); // one-dimension
448  typename View::GD_t::Vec_Xptr grid = {&times[0]}; // one set of pointers
449  gd.setx( gextents, grid );
450 
451  // set response-grid (multiple in this case)
452  gd.setr( prototype.rextents(), response );
453 
454  // interpolate (linear for now)
455  std::vector<float> time_wts( nx );
456  interpwts_lin( times, time, &time_wts );
457 
458  typename View::GD_t::Vec_Xptr wts = {&time_wts[0]};
459  typename View::GD_t::Vec_Z interp_view = gd.rgridsum( wts );
460 
461  // create new interpolated StateInfo by pushing interpolated
462  // response back to newly created state info
463  typename View::Type::SP interp = prototype.permute( interp_view );
464 
465  return interp;
466 }

487
