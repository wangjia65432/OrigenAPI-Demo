137 // n - number units
138 // m - mass units
139 // d - decay units
140 // are captured by the converter macro below and uses the mol

143 // NEW = OLD * (1/c_old) * c_new

192 double ConcentrationConverter::convert_to( ConcentrationUnit unew,
193  int izzzaaa,
194  ConcentrationUnit uold,
195  double val,
196  double volume ) const
197 {
198  // don't care about zeros
199  if( val == 0.0 ) return 0.0;
200 
201  // always allow this
202  if( uold == unew ) return val;
203 
204  // conversion matrix (see Polaris/FrontEnd/TimeUnit for better ways)
205  if( isNumberUnit( uold ) )
206  {
207  if( isNumberUnit( unew ) )
208  return n2n( unew, izzzaaa, uold, val, volume );
209  if( isMassUnit( unew ) ) return n2m( unew, izzzaaa, uold, val, volume );
210  if( isDecayUnit( unew ) )
211  return n2d( unew, izzzaaa, uold, val, volume );
212  }
213  else if( isMassUnit( uold ) )
214  {
215  if( isMassUnit( unew ) ) return m2m( unew, izzzaaa, uold, val, volume );
216  if( isNumberUnit( unew ) )
217  return m2n( unew, izzzaaa, uold, val, volume );
218  if( isDecayUnit( unew ) )
219  return m2d( unew, izzzaaa, uold, val, volume );
220  }
221  else if( isDecayUnit( uold ) )
222  {
223  if( isDecayUnit( unew ) )
224  return d2d( unew, izzzaaa, uold, val, volume );
225  if( isNumberUnit( unew ) )
226  return d2n( unew, izzzaaa, uold, val, volume );
227  if( isMassUnit( unew ) ) return d2m( unew, izzzaaa, uold, val, volume );
228  }
229

231  // it is with the Spectrum or Reaction unit categories.
232  Ensure( isSpectrumUnit( uold ) || isReactionUnit( uold ) );
233  Ensure( isSpectrumUnit( unew ) || isReactionUnit( unew ) );
234  return BAD_DOUBLE;
