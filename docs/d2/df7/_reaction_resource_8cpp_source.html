150 void ReactionResource::import_fission_yields( const YieldResource& yr )
151 {
152  SCOPED_TIMER( "ReactionResource.import_fission_yields" );
153 
154  // iterate through actinide parents
155  for( int k = 0; k < yr.num_fissionables(); ++k )
156  {
157  const YieldParent& fn = yr.fissionable_at( k );
158 
159  int izzzaaa = fn.id();
160 
161  // an empty yield nuclides means we take it all, but if it is
162  // non-empty, we only keep going if the value is true
163  if( d_yield_nuclides.size() != 0 )
164  {
165  YieldNuclides_t::const_iterator ity =
166  d_yield_nuclides.find( izzzaaa );
167  if( ity == d_yield_nuclides.end() ) continue;
168  if( !ity->second ) continue;
169  }
170 
171  Vec_Int fp_ids;
172  fn.get_fp_ids( &fp_ids );
173 
174  // data collection
175  Vec_Flt xvalues;
176  std::map<int, Vec_Flt> yvalues;
177 
178  // iterate through yield sets (usually 3)
179  for( int i = 0; i < fn.num_energies(); i++ )
180  {
181  const YieldData& data = fn.data_at( i );
182  float energy = data.energy();
183 
184  // get xvalues
185  xvalues.push_back( energy );
186 
187  Check( (int)fp_ids.size() == data.num_yields() );
188  for( int ifp = 0; ifp < data.num_yields(); ++ifp )
189  {
190  // get yvalues
191  int fp_iza = fp_ids[ifp];
192  yvalues[fp_iza].push_back( data.yield_at( ifp ) );
193  }
194  }
195 
196  // have collected all the data for this parent
197  // get the AmpxData container and create the interpolation functions
198  if( xvalues.size() > 0 )
199  {
200  Check( this->count( izzzaaa ) != 0 );
201  ReactionResource::Nuclide& a_nd = this->find( izzzaaa )->second;
202 
203  Check( a_nd.count( 18 ) != 0 );
204  ReactionResource::Nuclide::Xs& a_xs = a_nd.find( 18 )->second;
205 
206  Check( a_xs.yield_type !=
207  YIELD_ISOMERIC ); // means we already set it
208  a_xs.yield_type = YIELD_DISTRIBUTED;
209 
210  std::map<int, Vec_Flt>::const_iterator iv;
211  for( iv = yvalues.begin(); iv != yvalues.end(); ++iv )
212  {
213  int fp_iza = iv->first;
214  // Check(a_xs.iyield.count(fp_iza)!=0);
215  a_xs.iyield[fp_iza]. /* Interp1d<float> */
216  set_data( xvalues, iv->second ); /* yields */
217  }
218  }
219  }
220 
221  return;
222 }
