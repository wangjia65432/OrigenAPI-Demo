939  aj = di
940  num = nonp(i)

944  aj = aj + timefx*sum(ap(nul:num))
945  n = num + 1
946  end if
947  end if

950  if (aj > asumj) keepit = i

953  end if
954  nul = nonp(i) + 1
955  end do
