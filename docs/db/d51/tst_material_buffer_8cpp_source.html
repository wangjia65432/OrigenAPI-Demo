193 herr_t h5_write_units( hid_t dataset, const char* units )
194 {
195  // create a space padded string for 32 characters of units info
196  const size_t UNITS_SIZE( 32 );
197  std::string sunits( units );
198  for( size_t i = sunits.size(); i < UNITS_SIZE; ++i )
199  {
200  sunits += " ";
201  }
202 
203  // handle units
204  htri_t exists = H5Aexists( dataset, "units" );
205  herr_t status = 0;
206  if( !exists )
207  {
208  hid_t dataspace = H5Screate( H5S_SCALAR );
209  hid_t type = H5Tcopy( H5T_C_S1 );
210 
211  status = H5Tset_size( type, UNITS_SIZE + 1 );
212  Check( status >= 0 );
213 
214  status = H5Tset_strpad( type, H5T_STR_NULLTERM );
215  Check( status >= 0 );
216 
217  hid_t att = H5Acreate2(
218  dataset, "units", type, dataspace, H5P_DEFAULT, H5P_DEFAULT );
219 
220  status = H5Awrite( att, type, sunits.c_str() );
221  H5Tclose( type );
222  H5Aclose( att );
223  H5Sclose( dataspace );
224  }
225 
226  return status;
227 }
