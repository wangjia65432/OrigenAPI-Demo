57 ///
58 /// Load resources and create a TransitionSystem adding
59 /// additional user defined reactions.
60 ///
61 /// Additional reactions can be added by passing the user_* vectors.
62 ///
63 /// -"Adding" a reaction that exists in the MG data has no effect.
64 /// -This will never remove or alter existing reactions.
65 /// -All daughters of new reactions are produced to ground state.
66 /// -You cannot add reactions to nuclides that do not exist in the skeleton
67 /// library.
68 /// -If the daughter of a new reaction does not exist in the skeleton, it is
69 /// added to the TransitionSystem, but will not be carried over to
70 /// TransitionMatrix. The removal associated with such reactions _is_
71 /// carried over to the TransitionMatrix, but will remove the nuclide
72 /// without producing a daughter.
73 /// -Fission may be added, but will have zero yields.
74 ///
75 /// If fission is added for a nuclide that has no MG data but has yields in the
76 /// skeleton library, inconsistencies will happen. This is supposed to never
77 /// happen but there are some data concictency problems
78 ///
79 ///
80 /// An example of the user_* lists:
81 /// -user_nuclide_list = {92235, 1092235, 94239, 54135};
82 /// -user_num_mts = {2, 1, 0, 2};
83 /// -user_mt_list = {MT1_92235, MT2_92235, MT1_92235m, MT1_Xe135, MT2_Xe135};
84 ///
85 /// \param[in] library_defienition Filepath to a skeleton Origen library.
86 /// \param[in] aux_mg_library Filepath to a multi-group library.
87 /// \param[in] fission_yields Filepath to a fission yield library.
88 /// \param[in] user_nuclide_list List of nuclide identifiers for which to
89 /// add reactions in izzzaaa format.
90 /// \param[in] user_num_mts Number of reactions to add for each nuclide.
91 /// \param[in] user_mt_list List of the reaction MT numbers to add
92 /// indexed so that the reactions for the first
93 /// nuclide are first, then those for the second
94 /// and so on.
95 ///
96 //==============================================================================
97 void TransitionMatrixUpdater::load( std::string library_definition,
98  std::string aux_mg_library,
99  std::string fission_yields,
100  const std::vector<int>& user_nuclide_list,
101  const std::vector<int>& user_num_mts,
102  const std::vector<int>& user_mt_list )
103 {
104  Require( user_nuclide_list.size() == user_num_mts.size() );
105  Require( std::accumulate( user_num_mts.begin(), user_num_mts.end(), 0 ) ==
106  user_mt_list.size() );
107 
108  this->clear();
109 
110  // load up skeleton library (definition)
111  LibraryIO lib_io;
112  b_lib = Origen::SP_Library( new Origen::Library() );
113  bool success_lib = lib_io.load( *b_lib, library_definition, b_status );
114  if( !success_lib )
115  {
116  b_status.appendError( "Could not open library_definition=" +
117  library_definition +
118  " and therefore will not try to open "
119  "aux_mg_library or fission_yields" );
120  return;
121  }
122  else
123  {
124  b_status.clear();
125  }
126  jDebugLine( "post-lib" );
127 
128  // load up AMPX JEFF
129  ReactionResourceIO rr_io;
130  SP_ReactionResource sp_rr;
131  if( b_storage_bounds.size() != 0 )
132  {
133  sp_rr = std::make_shared<ReactionResource>( b_storage_bounds );
134  }
135  else
136  {
137  sp_rr = std::make_shared<ReactionResource>();
138  }
139  bool success_rr = rr_io.load( *sp_rr, aux_mg_library, b_status );
140  if( !success_rr )
141  {
142  b_status.appendError(
143  "Could not open aux_mg_library=" + aux_mg_library +
144  " and therefore will not try to open fission_yields" );
145  return;
146  }
147  else
148  {
149  b_status.clear();
150  }
151  jDebugLine( "post-rr" );
152 
153  // load up yields
154  YieldResourceIO yr_io;
155  YieldResource yr;
156  bool success_yr = yr_io.load( yr, fission_yields, b_status );
157  if( !success_yr )
158  {
159  b_status.appendError( "Could not open fission_yields=" +
160  fission_yields );
161  return;
162  }
163  else
164  {
165  b_status.clear();
166  }
167  jDebugLine( "post-yield with success=" << success_yr );
168 
169  // import stuff--in the future the library definition
170  // should influence what we import, but for now, just
171  // get everything
172  jDebugTaggedLine( "starting import of fission yields" );
173  sp_rr->import_fission_yields( yr );
174  b_tsa.set_reaction_resource( sp_rr );
175 
176  // >>> Add user requested additional nuclides and reactions.
177 
178  // Check that all nuclides exists in the Library
179  {
180  std::vector<int> user_to_master, master_to_user;
181  b_lib->definition().nuclide_set().mapping_sets(
182  user_nuclide_list, &user_to_master, &master_to_user );
183  for( size_t i = 0; i < user_nuclide_list.size(); ++i )
184  {
185  bool exists_in_lib = ( user_to_master[i] != -1 );
186  std::ostringstream ss;
187  ss << "Trying to add user-defined reaction to "
188  << "nuclide " << user_nuclide_list[i]
189  << " which does not exist in the skeleton Origen library "
190  << library_definition;
191  Insist( exists_in_lib, ss.str().c_str() );
192  }
193  }
194 
195  // Loop over user nuclides to add reactions.
196  int index_mt_first = 0;
197  for( int i = 0; i < user_nuclide_list.size(); ++i )
198  {
199  // If the nuclide already exists, this only returns the pointer.
200  Nuclide::SP nuc = b_ts.add_nuclide( user_nuclide_list[i] );
201 
202  // Loop over all user requested reactions for this nuclide.
203  for( int j = 0; j < user_num_mts[i]; j++ )
204  {
205  int mt = user_mt_list[index_mt_first + j];
206 
207  // If the reaction already exists this only returns the ptr.
208  Channel::SP channel = nuc->add_loss( NEUTRON_FIELD, mt, 0 );
209 
210  if( mt != 18 )
211  {
212  // Add production of daughter, always ground state.
213  // If the reaction, and hence the transition already
214  // exists this merely returns the pointer.
215  Transition::SP transition = channel->add_transition( 0 );
216  transition->set_yield( 1 );
217  }
218  }
219  index_mt_first += user_num_mts[i];
220  }
221 
222  // extend the transition system
223  b_tsa.extend( &b_ts );
224 
225  // get library components
226  const TransitionStructure& tstr = b_lib->transition_structure();
227  const LibraryHeader& lh = b_lib->definition();
228 
229  lh.nuclide_set().get_ids( b_sizzzaaa_list );
230 
231  // get skeleton transition matrix
232  b_trx = b_lib->newsp_transition_matrix_at( 0 );
233 
234  // nuclide sequence & reaction_gain_sequence are what we need to update
235  b_bld.search_reaction_transitions( &b_nuclide_sequence,
236  &b_reaction_gain_sequence,
237  tstr.num_decay_parents(),
238  tstr.num_parents(),
239  tstr.parent_positions(),
240  tstr.transition_ids(),
241  b_sizzzaaa_list,
242  b_ts );
243  b_initialized = true;
244 }

451 void TransitionMatrixUpdater::update_fission_absorption_xs(
452  std::string id_form,
453  const std::vector<int>& nuclide_list,
454  const Vec_Flt& fission_xs,
455  const Vec_Flt& absorption_xs )
456 {
457  jDebugTaggedLine(
458  "entering TransitionMatrixUpdater::update_fission_absorption_xs" );
459  Require( b_initialized );
460  Require( nuclide_list.size() == fission_xs.size() );
461  Require( fission_xs.size() == absorption_xs.size() );
462 
463  std::vector<SP_Nuclide> ts_nuclides;
464  fetch_ts_nuclides( &ts_nuclides, b_ts, id_form, nuclide_list );
465 
466  SP_Nuclide n;
467  SP_Channel c;
468  for( size_t i = 0; i < ts_nuclides.size(); ++i )
469  {
470  n = ts_nuclides[i];
471  double abs_xs = absorption_xs[i];
472  double fis_xs = fission_xs[i];
473  double cap_xs = abs_xs - fis_xs;
474 
475  jDebugLine( "updating nuclide=" << ( n == nullptr ? 0 : n->id() )
476  << " with abs_xs="
477  << abs_xs
478  << " fis_xs="
479  << fis_xs
480  << " cap_xs="
481  << cap_xs );
482 
483  if( abs_xs < 0.0 )
484  {
485  jDebugLine( " WARNING: " << ( n == nullptr ? 0 : n->id() )
486  << " with abs_xs="
487  << abs_xs
488  << " < 0"
489  << " will not be updated!" );
490  continue;
491  }
492  // Check(abs_xs>=0);
493  // Check(fis_xs>=0);
494  // Check(abs_xs>=fis_xs);
495 
496  if( n == nullptr )
497  {
498  jDebugLine( " ...failed to update id=" << nuclide_list[i] );
499  continue;
500  }
501 
502  std::vector<SP_Channel> losses = n->losses();
503  // infinitely dilute capture xs
504  std::vector<double> id_cap_xs( losses.size(), 0.0 );
505  double total_id_cap_xs = 0.0;
506  for( size_t j = 0; j < losses.size(); ++j )
507  {
508  int reaction_mt = losses[j]->id();
509  // set fission
510  if( reaction_mt == 18 )
511  {
512  losses[j]->set_xs( fis_xs );
513  }
514  // sum the remainder
515  else if( reaction_mt >= 102 && reaction_mt <= 117 )
516  {
517  double this_cap = losses[j]->xs();
518  id_cap_xs[j] = this_cap;
519  total_id_cap_xs += this_cap;
520  }
521  }
522  jDebugLine( " infinitely-dilute cap_xs(" << n->id() << ")="
523  << total_id_cap_xs );
524 
525  if( total_id_cap_xs > 0.0 )
526  {
527  // calculate self-shielding factor for capture
528  double f_ss_cap = cap_xs / ( total_id_cap_xs );
529  jDebugLine(
530  " self-shielding factor(" << n->id() << ")=" << f_ss_cap );
531  // Check(cap_xs<=total_id_cap_xs); //self-shielding yields lower xs
532 
533  // loop again, this time modifying all the capture xs
534  for( size_t j = 0; j < losses.size(); ++j )
535  {
536  int reaction_mt = losses[j]->id();
537  if( reaction_mt >= 102 && reaction_mt <= 117 )
538  {
539  losses[j]->set_xs( id_cap_xs[j] * f_ss_cap );
540  continue;
541  }
542  Check( id_cap_xs[j] == 0.0 );
543  }
544  }
545  }
546  jDebugTaggedLine(
547  "leaving TransitionMatrixUpdater::update_fission_absorption_xs" );
548 }
