236  int const* ids,
237  double const* vals,
238  double wt )
239 {

241

271 bool Concentrations::accumulate_concs( const Concentrations& concs, double wt )
272 {
273  // create a new vals with units converted
274  Vec_Dbl vals( concs.vals_size() );
275  concs.convert_units( this->units(), &vals );
276 
277  // if volumetric we want result of this operation to be average
278  // and not sum.
279  bool volumetric = isVolumetric( this->units() );
280  if( volumetric )
281  {
282  // scale up by volumes
283  for( size_t i = 0; i < vals.size(); ++i ) vals[i] *= concs.volume();
284  for( size_t i = 0; i < b_vals->size(); ++i ) ( *b_vals )[i] *= b_volume;
285  }
286 
287  // increase this volume by weight
288  b_volume += concs.volume() * wt;
289 
290  bool all_found = accumulate_concs( concs.nuclide_set().ids(), vals, wt );
291  // reconstruct average
292  if( volumetric )
293  {
294  for( size_t i = 0; i < b_vals->size(); ++i ) ( *b_vals )[i] /= b_volume;
295  }
296  return all_found;

306 bool Concentrations::convert_units( ConcentrationUnit units,
307  double volume,
308  Vec_Dbl* vals ) const
309 {
310  Require( vals != nullptr );
311  ConcentrationConverter converter;
312  bool convertible = true;
313  converter.set_decay_resource( r_decay_resource );
314  converter.set_nuclide_resource( r_nuclide_resource );
315  vals->resize( b_vals->size() );
316  for( size_t i = 0; i < b_vals->size(); ++i )
317  {
318  // if we start with garbage, don't even try to convert
319  if( b_vals->at( i ) == BAD_DOUBLE )
320  {
321  convertible = false;
322  vals->at( i ) = b_vals->at( i );
323  continue;
324  }
325  else if( b_vals->at( i ) == 0.0 )
326  {
327  vals->at( i ) = 0.0;
328  // don't waste time on conversion
329  continue;
330  }
331 
332  vals->at( i ) =
333  converter.convert_to( units,
334  b_nuclide_set->izzzaaa_at( i ),
335  b_units,
336  b_vals->at( i ),
337  volume > 0.0 ? volume : b_volume );
338 
339  if( vals->at( i ) == BAD_DOUBLE ) convertible = false;
340  }
