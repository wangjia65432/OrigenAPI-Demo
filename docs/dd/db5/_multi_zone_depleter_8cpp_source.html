137 {

324 void MultiZoneDepleter::pre_transmute()
325 {
326  // num substeps
327  size_t ns = b_trel.size() - 1;
328 
329  // reset basis ids
330  // default is true (all materials in basis) if basis_ids are empty
331  // otherwise let basis_ids dictate and therefore default is false
332  bool def = true;
333  if( b_basis_ids.size() > 0 ) def = false;
334  b_is_basis.clear();
335  b_is_basis.resize( b_mats->size(), def );
336 
337  // reset scale
338  b_scaling.clear();
339  b_scaling.resize( ns + 1, 1.0 );
340 
341  // make sure all the basis ids get distributed
342  Remember( std::set<int> basis_ids( b_basis_ids ) );
343 
344  size_t i = 0;
345  for( auto it = b_mats->begin(); it != b_mats->end(); ++it )
346  {
347  // base variables
348  SP_Material mat = it->second;
349  int id = it->first;
350  jDebugLine( " checking material i=" << i << " name=" << mat->name()
351  << " id="
352  << id );
353 
354  // verify sizes
355  Check( mat->good() );
356 
357  // all materials must be updated in lock step
358  jDebugLine(
359  " checking material dt=" << mat->dt() << " global dt=" << b_dt );
360  Require( ScaleSTL::approx_eq( mat->dt(), b_dt ) );
361 
362  // basis
363  auto idf = b_basis_ids.find( id );
364  if( idf != b_basis_ids.end() )
365  {
366  Remember( basis_ids.erase( idf ) );
367  b_is_basis[i] = true;
368  }
369 
370  // initialize one of the history variables for each material by handling
371  // 4 combinations of global constraints and what was provided on
372  // material
373  double param = mat->param_hist( 0.0 );
374 
375  // SET T=0 POWER HISTORY
376  if( b_constraint_type == Material::POWER )
377  {
378  if( mat->flux_parameter() == Material::POWER )
379  {
380  mat->set_power_hist( 0.0, param );
381  }
382  else if( mat->flux_parameter() == Material::FLUX )
383  {
384  mat->set_power_hist( 0.0, param * mat->power_factor_bos() );
385  }
386  }
387  // SET T=0 FLUX HISTORY
388  else if( b_constraint_type == Material::FLUX )
389  {
390  if( mat->flux_parameter() == Material::POWER )
391  {
392  mat->set_flux_hist( 0.0, param / mat->power_factor_bos() );
393  }
394  else if( mat->flux_parameter() == Material::FLUX )
395  {
396  mat->set_flux_hist( 0.0, param );
397  }
398  }
399 
400  // increment
401  ++i;
402  }
403 
404  Ensure( basis_ids.empty() );

491 double MultiZoneDepleter::global_scaling_factor( size_t s, size_t itran )
492 {
493  jDebugLine( "##MultiZoneDepleter::global_scaling_factor(" << s << ","
494  << itran
495  << ")" );
496  Require( s < b_trel.size() );
497  Require( b_mats->size() == b_is_basis.size() );
498 
499  // choose beginning or end-of-step for constraint relative time
500  double trel = b_trel[s];
501  double global_constraint = constraint_hist( trel );
502  double global_sum = 1.0;
503 
504  // handle power constraint
505  if( global_constraint == 0.0 || !has_constraint() )
506  {
507  global_sum = 1.0;
508  }
509  else if( b_constraint_type == Material::POWER )
510  {
511  double cmat = 0.0; // local value for constraint
512  size_t i = 0;
513  for( auto it = b_mats->begin(); it != b_mats->end(); ++it )
514  {
515  SP_Material mat = it->second;
516  if( mat->is_decay() )
517  {
518  ++i;
519  continue;
520  }
521  if( b_is_basis[i] )
522  {
523  double ph = mat->power_hist( trel );
524  cmat += ph;
525  }
526  ++i;
527  }
528  double ctot = cmat + b_bg;
529 
530  // global communication
531  Vec_Dbl cx( {ctot} );
532  allreduce( cx );
533 
534  global_sum = cx[0];
535  }
536 
537  // handle flux constraint
538  else if( b_constraint_type == Material::FLUX )
539  {
540  double cmat = 0.0;
541  double vmat = 0.0;
542  size_t i = 0;
543  for( auto it = b_mats->begin(); it != b_mats->end(); ++it )
544  {
545  SP_Material mat = it->second;
546  if( mat->is_decay() )
547  {
548  ++i;
549  continue;
550  }
551  if( b_is_basis[i] )
552  {
553  vmat += mat->volume_hist( trel );
554  cmat += mat->flux_hist( trel );
555  }
556  ++i;
557  }
558 
559  // constraint / ( cmat*volm/(volm+volbg) + cbg*volbg/(volm+volbg) )
560  double vtot = b_bg_volume + vmat;
561  double ctot = cmat * vmat + b_bg * b_bg_volume;
562 
563  // global communication
564  Vec_Dbl cx( {ctot, vtot} );
565  allreduce( cx );
566 
567  // volume should be >0
568  Check( cx[1] > 0.0 );
569 
570  global_sum = cx[0] / cx[1];
571  }
572 
573  double scale = 1.0;
574  if( global_constraint > 0.0 )
575  {
576  Check( global_sum > 0.0 );
577  scale = global_constraint / global_sum;
578  }
579 
580  return scale;
581 }

587 bool MultiZoneDepleter::execute()
588 {
589  jDebugLine( "##MultiZoneDepleter::execute()" );
590 
591  // setup for transmutation
592  pre_transmute();
593 
594  // perform an initial scaling at s=0 itran=0
595  b_scaling[0] = global_scaling_factor( 0, 0 );
596  Check( b_constraint_its_max > 0 );
597  for( size_t s = 0; s < b_trel.size() - 1; ++s )
598  {
599  jDebugLine( " substep(" << s << ") with trel0=" << b_trel[s]
600  << " --> trel1="
601  << b_trel[s + 1]
602  << " old scaling s0="
603  << b_scaling[s] );
604 
605  // enter constraint iteration loop
606  // if no constraints, we pop out after one iteration
607  double last_scaling = b_scaling[s];
608  size_t itran = 0;
609  double relchange = 1.0;
610  while( itran < b_constraint_its_max )
611  {
612  // transmute
613  transmute( s, itran );
614  jDebugLine( " transmute(" << s << "," << itran
615  << ") complete." );
616 
617  // calculate new scaling factor
618  b_scaling[s + 1] = global_scaling_factor( s + 1, itran );
619  jDebugLine( " new scaling s1(" << s << "," << itran << ")="
620  << b_scaling[s + 1] );
621 
622  // check tolerance
623  relchange = std::fabs( b_scaling[s + 1] / last_scaling - 1.0 );
624  jDebugLine( " new/old-1(" << s << "," << itran << ")/tol="
625  << relchange
626  << "/"
627  << b_constraint_its_tol );
628 
629  // final check
630  if( relchange <= b_constraint_its_tol ) break;
631 
632  // prepare for next iteration
633  last_scaling = b_scaling[s + 1];
634  ++itran;
635  }
636  Check( !has_constraint() ? itran == 0 : itran <= b_constraint_its_max );
637  Check( !has_constraint() ? relchange == 0.0 : true );
638 
639  jDebugLine( " substep(" << s << ") complete with " << itran + 1
640  << " transmutes." );
641  }
642 
643  // wrapup operations
644  post_transmute();
645 
646  return b_status->numError() == 0;
647
