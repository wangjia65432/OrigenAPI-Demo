245 bool Nuclide::good_impl( Vec_Str& errors ) const
246 {
247  bool pass = Checkable::good_impl( errors );
248  if( !has_initialized_species() )
249  {
250  ORIGEN_CHECKABLE_ERROR(
251  name() << " has uninitialized species", pass, errors );
252  }
253  else
254  {
255  if( species()->mass_number() <= 0 )
256  ORIGEN_CHECKABLE_ERROR(
257  name() << " has species with mass_number<=0", pass, errors );
258  if( species()->charge_number() < 0 )
259  ORIGEN_CHECKABLE_ERROR(
260  name() << " has species with charge_number<0", pass, errors );
261  }
262  if( isomeric_state() < 0 )
263  {
264  ORIGEN_CHECKABLE_ERROR(
265  name() << " has isomeric_state<0 ", pass, errors );
266  }
267  if( !has_initialized_system() )
268  ORIGEN_CHECKABLE_ERROR(
269  name() << " has uninitialized system", pass, errors );
270 
271  // Checks if two gains have same target AND channel
272  std::vector<Transition::WP> gains = this->gains();
273  std::set<std::pair<int, int>> found;
274  for( size_t i = 0; i < gains.size(); ++i )
275  {
276  std::pair<int, int> cp_id( gains[i]->channel()->id(),
277  gains[i]->channel()->parent()->id() );
278  if( found.count( cp_id ) > 0 )
279  {
280  ORIGEN_CHECKABLE_ERROR(
281  name() << " has two or more instances of channel="
282  << gains[i]->channel()->name()
283  << " parent="
284  << gains[i]->channel()->parent()->name(),
285  pass,
286  errors );
287  }
288  else
289  {
290  found.insert( cp_id );
291  }
292  }
293 
294  return pass;
295 }
