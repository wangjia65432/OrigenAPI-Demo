699  {

701  }
702  sig /= sumflux;
703  jDebugTaggedLine( "update_by_mix: updating channel xs for nuclide="

711  auto direct = yield_map.find( mt );
712  if( direct != yield_map.end() )
713  {
714  auto nu = direct->second.first;
715  auto transition = direct->second.second;
716  float yield = 0;
717  // NOTE: The treatment of nubar below does not allow us to
718  // use the Aux library (JEFF) nubar with AMPX fission xs.
719  // It is complicated because the JEFF nubar and fisxs could
720  // be stored in a different group structure.
721  for( size_t g = 0; g < flux.size(); ++g )
722  {
723  yield += nu->getValues()[g] * flux[g] * xs->getValues()[g];
724  }
725  yield /= ( sumflux * sig );
726  jDebugTaggedLine(
727  "update_by_mix: updating transition yield xs for nuclide="
728  << izzzaaa
729  << " mt="
730  << mt
731  << " yield="
732  << yield );
733  transition->set_yield( yield );
734  }
735  }
736 
737  jDebugTaggedLine(
738  "update_by_mix: finished nuclide izzzaaa=" << izzzaaa );
739  }
