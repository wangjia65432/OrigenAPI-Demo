1412 !
1413 ! First step: Terminate PHASE3 of the search for an on-scale step size.
1414 ! The step size is not on scale, so ERR may not be accurate;
1415 ! reduce H by a fixed factor. Failed attempts to take the
1416 ! first step are not counted.
1417 ! Later step: Use ERR to compute an "optimal" reduction of H. More than
1418 ! one failure indicates a difficulty with the problem and an
1419 ! ERR that may not be accurate, so reduce H by a fixed factor.
1420 !
1421  if (comm%at_t_start) then
1422  phase3 = .false.; alpha = comm%rs1
1423  else
1424  comm%bad_step_count = comm%bad_step_count + 1
1425  comm%stiff_bad_step_count = comm%stiff_bad_step_count + 1
1426  if (failed) then
1427  alpha = comm%rs1
1428  else
1429  alpha = comm%safety*(comm%tol/err)**comm%expon
1430  alpha = max(alpha,comm%rs1)
1431  end if
1432  end if
1433  comm%h = alpha*comm%h; failed = .true.; cycle take_step
1434  end if
1435 !
1436 ! Successful step.
1437 !
1438 ! Predict a step size appropriate for the next step. After the first
1439 ! step the prediction can be refined using an idea of H.A. Watts that
1440 ! takes account of how well the prediction worked on the previous step.
1441 !
1442  beta = (err/comm%tol)**comm%expon
1443  if (.not.comm%at_t_start) then
1444  t1 = (err**comm%expon)/comm%h
1445  t2 = (comm%errold**comm%expon)/comm%h_old
1446  if (t1<t2*hundrd .and. t2<t1*hundrd) beta = beta*(t1/t2)
1447  end if
1448  alpha = comm%rs3
1449  if (comm%safety < beta*alpha) alpha = comm%safety/beta
1450 !
1451 ! On the first step a search is made for an on-scale step size. PHASE2
1452 ! of the scheme comes to an end here because a step size has been found
1453 ! that is both successful and has a credible local error estimate. Except
1454 ! in the special case that the first step is also the last, the step is
1455 ! repeated in PHASE3 as long as an increase greater than RS2 appears
1456 ! possible. An increase as big as RS3 is permitted. A step failure
1457 ! terminates PHASE3.
1458 !
1459  if (comm%at_t_start) then
1460  comm%phase2 = .false.
1461  phase3 = phase3 .and. .not. comm%at_t_end .and. (alpha > comm%rs2)
1462  if (phase3) then
1463  comm%h = alpha*comm%h; cycle take_step
1464  end if
1465  end if
1466 !
1467 ! After getting on scale, step size changes are more restricted.
1468 !
1469  alpha = min(alpha,comm%rs)
1470  if (failed) alpha = min(alpha,one)
1471  alpha = max(alpha,comm%rs1)
1472  comm%h_old = comm%h; comm%h = alpha*comm%h
1473 !
1474 ! For the diagnosis of stiffness, an average accepted step size, H_AVERAGE,
1475 ! must be computed.
1476 !
1477  if (comm%at_t_start) then
1478  comm%h_average = comm%h_old
1479  else
1480  comm%h_average = pt9*comm%h_average + pt1*comm%h_old
1481  end if
1482 !
1483  comm%at_t_start = .false.; comm%errold = err; comm%t_old = comm%t
1484 !
1485 ! Take care that T is set to precisely T_END when the end of the
1486 ! integration is reached.
1487 !
1488  if (comm%at_t_end) then
1489  comm%t = comm%t_end
1490  else
1491  comm%t = comm%t + comm%h_old
1492  end if
1493 !
1494 ! Increment counter on accepted steps. Note that successful steps
1495 ! that are repeated whilst getting on scale are not counted.
1496 !
1497  comm%step_count = comm%step_count + 1
1498 !
1499 ! Advance the current solution and its derivative. Note that the previous
1500 ! derivative will overwrite Y_NEW (see pointer assignments in SETUP).
1501 !
1502  comm%y_old = comm%y; comm%y = comm%y_new
1503  comm%yp_old = comm%yp
1504 !
1505  if (comm%fsal) then
1506 !
1507 ! When FSAL = .TRUE., YP is the last stage of the step.
1508 !
1509  comm%yp = comm%stages(:,comm%last_stage)
1510  else
1511 !
1512 ! Call F to evaluate YP.
1513 !
1514  comm%yp = f(comm%t,comm%y); comm%f_count = comm%f_count + 1
1515  end if
1516 !
1517 ! If global error assessment is desired, advance the secondary
1518 ! integration from TOLD to T.
1519 !
1520  if (comm%erason) then
1521  call truerr_r1(comm,f,ier)
1522  if (ier==6) then
1523 !
1524 ! The global error estimating procedure has broken down. Treat it as a
1525 ! failed step. The solution and derivative are reset to their values at
1526 ! the beginning of the step since the last valid error assessment refers
1527 ! to them.
1528 !
1529  comm%step_count = comm%step_count - 1; comm%erasfl = .true.
1530  comm%at_t_end = .false.
1531  comm%t = comm%t_old; comm%h = comm%h_old
1532  comm%y = comm%y_old; comm%yp = comm%yp_old
1533  if (comm%step_count > 0) then
1534  nrec = 2; write (comm%rec,"(a/a,e13.5/a)") &
1535 " ** The global error assessment may not be reliable for T past ",&
1536 " ** T_NOW = ",comm%t,". The integration is being terminated."
1537  exit body
1538  else
1539  nrec = 2; write (comm%rec,"(a/a)") &
1540 " ** The global error assessment algorithm failed at the start of the ",&
1541 " ** integration. The integration is being terminated."
1542  exit body
1543  end if
1544  end if
1545  end if
1546  exit take_step
1547  end do take_step
1548  exit body
1549 end do body
1550 !
1551 ! Exit point for STEP_INTEGRATE
1552 ! Set the output variables and flag that interpolation is permitted
1553 !
1554 if (ier < fatal) then
1555  t_now = comm%t; comm%at_t_end = t_now == comm%t_end
1556  comm%chkeff = comm%at_t_end;
1557  y_now = comm%y; yderiv_now = comm%yp
1558  comm%ymax = max(abs(comm%y),comm%ymax)
1559  if (ier==just_fine) then
1560  state = usable; call set_saved_state_r1("INTERPOLATE",state,comm)
1561  end if
1562 end if
1563 !
1564 ! Call RKMSG_R1 to report what happened and set FLAG.
1565 !
1566 call rkmsg_r1(ier,srname,nrec,comm,flag)
1567 !
1568 end subroutine step_integrate_r1

1571 subroutine truerr_r1(comm,f,ier)
1572 !
1573 ! Part of rksuite_90 v1.0 (Aug 1994)
1574 ! software for initial value problems in ODEs
1575 !
1576 ! Authors: R.W. Brankin (NAG Ltd., Oxford, England)
1577 ! I. Gladwell (Math Dept., SMU, Dallas, TX, USA)
1578 ! see main doc for contact details
1579 !
1580 type(rk_comm_real_1d), intent(inout) :: comm
1581 integer, intent(inout) :: ier
1582 !
1583 interface
1584  function f(t,y)
1585  use rksuite_90_prec, only:wp
1586  real(kind=wp), intent(in) :: t !indep!
1587  real(kind=wp), dimension(:), intent(in) :: y !dep!
1588  real(kind=wp), dimension(size(y,1)) :: f !dep!
1589  end function f
1590 end interface
1591 !
1592 real(kind=wp) :: hmin, hsec !indep!
1593 real(kind=wp) :: diff, errmax, mxerlc, tsec, ge_err, ge_test1, ge_test2
1594 integer :: istep, level
1595 !
1596 integer, parameter :: just_fine=1
1597 real(kind=wp), parameter :: pt1=0.1_wp, ten=10.0_wp
1598 real(kind=wp), dimension(:,:), pointer :: ge_stages !dep!
1599 real(kind=wp), dimension(:), pointer :: ge_y, ge_yp, ge_y_new !dep!
1600 real(kind=wp), dimension(:), pointer :: ge_err_estimates, y !dep!
1601 real(kind=wp), dimension(:), pointer :: ge_assess, weights !shp-dep!
1602 !
1603 ge_stages => comm%ge_stages
1604 ge_y => comm%ge_y
1605 ge_yp => comm%ge_yp
1606 ge_y_new => comm%ge_y_new
1607 ge_err_estimates => comm%ge_err_estimates
1608 ge_assess => comm%ge_assess
1609 y => comm%y
1610 weights => comm%weights
1611 !
1612 tsec = comm%t - comm%h_old
1613 hsec = comm%h_old/real(comm%no_of_ge_steps,kind=wp)
1614 hmin = max(comm%sqtiny,comm%toosml*max(abs(tsec),abs(comm%t)))
1615 body: do
1616  if (abs(hsec)<hmin) then
1617  ier = 6; exit body
1618  end if
1619  ge_test1 = comm%tol/real(comm%no_of_ge_steps,kind=wp)
1620  ge_test2 = comm%tol/ten; level = 0
1621 !
1622 ! The subroutine STEP is used to take a step.
1623 !
1624 ! Perform secondary integration.
1625 !
1626  do istep = 1, comm%no_of_ge_steps
1627 !
1628 ! Take a step.
1629  call step_r1(comm,f,tsec,ge_y,ge_yp,ge_stages,ge_test1,hsec,ge_y_new, &
1630  ge_err_estimates,ge_err)
1631 !
1632 ! The primary integration is using a step size of H_OLD and the
1633 ! secondary integration is using the smaller step size
1634 ! HSEC = H_OLD/(NO_OF_GE_STEPS).
1635 ! If steps of this size were taken from the same starting point and the
1636 ! asymptotic behavior were evident, the smaller step size would result in
1637 ! a local error that is considerably smaller, namely by a factor of
1638 ! 1/(NO_OF_GE_STEPSSEC**(ORDER+1)). If the two approximate solutions are
1639 ! close and TOL is neither too large nor too small, this should be
1640 ! approximately true. The step size is chosen in the primary integration
1641 ! so that the local error ERR is no larger than TOL. The local error,
1642 ! GE_ERR, of the secondary integration is compared to TOL in an attempt to
1643 ! diagnose a secondary integration that is not rather more accurate than
1644 ! the primary integration.
1645 !
1646  if (ge_err>=ge_test1) then
1647  level = 2
1648  else if (ge_err>ge_test2) then
1649  level = level + 1
1650  end if
1651  if (level>=2) then
1652  ier = 6; exit body
1653  end if
1654 !
1655 ! Advance TSEC and the dependent variables GE_Y and GE_YP.
1656 !
1657  tsec = comm%t - real(comm%no_of_ge_steps-istep,kind=wp)*hsec
1658  ge_y = ge_y_new
1659 !
1660  if (comm%fsal) then
1661 !
1662 ! When FSAL = .TRUE., the derivative GE_YP is the last stage of the step.
1663 !
1664  ge_yp = ge_stages(:,comm%last_stage)
1665  else
1666 !
1667 ! Call F to evaluate GE_YP.
1668 !
1669  ge_yp = f(tsec,ge_y); comm%ge_f_count = comm%ge_f_count + 1
1670  end if
1671 !
1672  end do
1673 !
1674 ! Update the maximum error seen, GE_MAX_CONTRIB, and its location,
1675 ! T_GE_MAX_CONTRIB. Use local variables ERRMAX and MXERLC.
1676 !
1677  errmax = comm%ge_max_contrib; mxerlc = comm%t_ge_max_contrib;
1678  diff = maxval(abs(ge_y-y)/weights) !spec-ar!
1679  if (diff>errmax) then
1680  errmax = diff; mxerlc = comm%t
1681  end if
1682 !
1683 ! If the global error is greater than 0.1, the solutions have diverged so
1684 ! far that comparing them may not provide a reliable estimate of the global
1685 ! error. The test is made before GE_ASSESS and GE_MAX_CONTRIB,
1686 ! T_GE_MAX_CONTRIB are updated so that on a failure, they refer to the
1687 ! last reliable results.
1688 !
1689  if (errmax>pt1) then
1690  ier = 6
1691  else
1692  comm%ge_max_contrib = errmax; comm%t_ge_max_contrib = mxerlc;
1693  ge_assess = ge_assess + (abs(ge_y-y)/weights)**2
1694  ier = just_fine
1695  end if
1696  exit body
1697 !
1698 end do body
1699 !
1700 end subroutine truerr_r1

1702 subroutine step_r1(comm,f,tnow,y,yp,stages,tol,htry,y_new, &
1703  errest,err,hmin,phase_2)
1704 !
1705 ! Part of rksuite_90 v1.0 (Aug 1994)
1706 ! software for initial value problems in ODEs
1707 !
1708 ! Authors: R.W. Brankin (NAG Ltd., Oxford, England)
1709 ! I. Gladwell (Math Dept., SMU, Dallas, TX, USA)
1710 ! see main doc for contact details
1711 !
1712 type(rk_comm_real_1d), intent(inout), target :: comm
1713 real(kind=wp), intent(out) :: err
1714 real(kind=wp), intent(inout) :: htry !indep!
1715 real(kind=wp), intent(in) :: tnow !indep!
1716 real(kind=wp), intent(in) :: tol
1717 real(kind=wp), intent(in), optional :: hmin !indep!
1718 logical, intent(inout), optional :: phase_2
1719 !
1720 real(kind=wp), dimension(:), intent(in) :: y, yp !dep!
1721 real(kind=wp), dimension(:), intent(out) :: errest, y_new !dep!
1722 real(kind=wp), dimension(:,:), intent(out) :: stages !dep!
1723 !
1724 interface
1725  function f(t,y)
1726  use rksuite_90_prec, only:wp
1727  real(kind=wp), intent(in) :: t !indep!
1728  real(kind=wp), dimension(:), intent(in) :: y !dep!
1729  real(kind=wp), dimension(size(y,1)) :: f !dep!
1730  end function f
1731 end interface
1732 !
1733 real(kind=wp) :: tstg !indep!
1734 integer :: i, j
1735 logical :: cutbak, main
1736 !
1737 intrinsic abs, max, sign
1738 !
1739 real(kind=wp), dimension(:), pointer :: weights, thresh !shp-dep!
1740 real(kind=wp), dimension(:,:), pointer :: a !real!
1741 real(kind=wp), dimension(:), pointer :: b, bhat, c !real!
1742 integer, dimension(:), pointer :: ptr !integer!
1743 !
1744 real(kind=wp), parameter :: zero=0.0_wp, half=0.5_wp, one=1.0_wp
1745 !
1746 ! ERREST is used for working storage in this computation.
1747 !
1748 weights => comm%weights
1749 thresh => comm%thresh
1750 a => comm%a
1751 b => comm%b
1752 bhat => comm%bhat
1753 c => comm%c
1754 ptr => comm%ptr
1755 !
1756 main = present(hmin) .and. present(phase_2)
1757 attempt_step: do
1758 !
1759  if (main) then
1760  if (comm%phase2) weights = max(thresh,abs(y))
1761  end if
1762 !
1763  do i = 2, comm%no_of_stages
1764  errest = a(i,1)*yp
1765  do j = 2, i - 1
1766  if (a(i,j)/=zero) errest = errest + a(i,j)*stages(:,ptr(j))
1767  end do
1768  y_new = y + htry*errest
1769 !
1770 ! METHOD = 'M' is special in that an estimate of the local error can be
1771 ! formed before the step is completed. If the step is a failure,
1772 ! return immediately. Otherwise, complete the step and compute a more
1773 ! accurate error estimate.
1774 !
1775  if (comm%rk_method==2 .and. i==7) then
1776  call stepb
1777  if (err>tol) return
1778  end if
1779 !
1780  tstg = tnow + c(i)*htry
1781  if (main .and. comm%at_t_end .and. c(i)==one) tstg = comm%t_end
1782  stages(:,ptr(i)) = f(tstg,y_new)
1783 !
1784 ! Increment the counter for the number of function evaluations
1785 ! depending on whether the primary or secondary integration is taking
1786 ! place.
1787 !
1788  if (main) then
1789  comm%f_count = comm%f_count + 1
1790  else
1791  comm%ge_f_count = comm%ge_f_count + 1
1792  end if
1793 !
1794 ! When PHASE2 is .TRUE. we are in the second phase of the automatic
1795 ! selection of the initial step size. The results of the first three
1796 ! stages are monitored in the subroutine STEPA for evidence that H is
1797 ! too large -- instability and/or an unreliable estimate of the error
1798 ! of the step is then possible. When the subroutine believes H to be
1799 ! too large, it returns CUTBAK = .TRUE. and a suitably reduced H for
1800 ! another try.
1801 !
1802  if (main) then
1803  if (phase_2) then
1804  if (i<=3 .and. abs(htry)>hmin) then
1805  call stepa(stages(:,ptr(i)),htry,cutbak)
1806  if (cutbak) then
1807  comm%at_t_end = .false.
1808 !
1809 ! Make sure that STEPA does not reduce the step size below the
1810 ! minimum. If it does, reset H to HMIN and deactivate PHASE2.
1811 !
1812  if (abs(htry)<=hmin) then
1813  htry = sign(hmin,htry); comm%phase2 = .false.
1814  end if
1815  cycle attempt_step
1816  end if
1817  end if
1818  end if
1819  end if
1820 !
1821  end do
1822 !
1823 ! Some formulas are constructed so that the last stage represents
1824 ! the result of the step (FSAL=.TRUE.), hence if the step is acceptable,
1825 ! it will be the first stage for the next step. When FSAL=.FALSE., we
1826 ! have to complete the computation of the step.
1827 !
1828  if (.not.comm%fsal) then
1829  errest = bhat(1)*yp
1830  do i = 2, comm%no_of_stages
1831  if (bhat(i)/=zero) errest = errest + bhat(i)*stages(:,ptr(i))
1832  end do
1833  y_new = y + htry*errest
1834  end if
1835 !
1836 ! Form an estimate of the error in the lower order formula by comparing
1837 ! it to the higher order formula of the pair. ERREST has been used
1838 ! as working storage above. The higher order approximation has been
1839 ! formed as Y_NEW = Y + HTRY*ERREST where ERREST is a linear
1840 ! combination of the stages of the formula. The lower order result also
1841 ! has the form Y plus HTRY times a different linear combination of
1842 ! the stages. Hence, this different linear combination of stages for
1843 ! the lower order formula can just be subtracted from the combination
1844 ! stored in ERREST to produce the errors. The result is then
1845 ! multiplied by HTRY to obtain the error estimate.
1846 !
1847  if (b(1)/=zero) errest = errest - b(1)*yp
1848  do i = 2, comm%no_of_stages
1849  if (b(i)/=zero) errest = errest - b(i)*stages(:,ptr(i))
1850  end do
1851  errest = htry*errest
1852 !
1853 ! The error in a solution component is measured relative to a weight
1854 ! that is the larger of a threshold and the size of the solution over
1855 ! the step. Using the magnitude of a solution component at both ends
1856 ! of the step in the definition of "size" increases the robustness of
1857 ! the test. When global error estimation is specified, the weight
1858 ! vector WEIGHTS is defined by the primary integration and is then
1859 ! used in the secondary integration.
1860 !
1861  if (main) weights = max(half*(abs(y)+abs(y_new)),thresh)
1862 !
1863  err = maxval(abs(errest/weights)) !spec-ar!
1864 !
1865  exit attempt_step
1866 !
1867 end do attempt_step
1868 !
1869 contains
1870 !
1871  subroutine stepa(ypnew,htry,cutbak)
1872 !
1873  real(kind=wp), intent(inout) :: htry !indep!
1874  real(kind=wp), dimension(:), intent(in) :: ypnew !dep!
1875  logical, intent(out) :: cutbak
1876 !
1877  real(kind=wp) :: argdif, fdiff, scl, tdiff, twt, ynrm, ystgnm
1878 !
1879 ! Update the weights to account for the current intermediate solution
1880 ! approximation Y_NEW. Compute the sizes of Y and Y_NEW in the
1881 ! new norm. The size of the Lipschitz constant is assessed by a difference
1882 ! in the arguments Y, Y_NEW and a difference in the function evaluated
1883 ! at these arguments.
1884 !
1885  weights = max(weights,abs(y_new))
1886  ynrm = maxval(abs(y)/weights) !spec-ar!
1887  ystgnm = maxval(abs(y_new)/weights) !spec-ar!
1888  argdif = maxval(abs(y_new-y)/weights) !spec-ar!
1889  fdiff = maxval(abs(ypnew-yp)/weights) !spec-ar!
1890 !
1891 ! The transformation of the equation to autonomous form is done
1892 ! implicitly. The difference of the arguments must take into account
1893 ! the difference between the values of the independent variable T and
1894 ! TSTG. The difference of the corresponding component of the function
1895 ! is zero because of the way the standard transformation is done.
1896 !
1897  tdiff = tstg - tnow
1898  twt = abs(comm%t_end-tnow)
1899  ynrm = max(ynrm,abs(tnow)/twt)
1900  ystgnm = max(ystgnm,abs(tstg)/twt)
1901  argdif = max(argdif,abs(tdiff)/twt)
1902 !
1903 ! The ratio FDIFF/ARGDIF is a lower bound for, and an approximation to,
1904 ! a Lipschitz constant L for the differential equation written in
1905 ! autonomous form. First we must ask if the difference ARGDIF is
1906 ! significant in the precision available. If it appears to be, we insist
1907 ! that abs(HTRY)*L be less than an approximate radius, STABILITY_RADIUS,
1908 ! of the stability region of the method. This is more stringent than
1909 ! necessary for stability, possibly a lot more stringent, but the aim is
1910 ! to get an HTRY small enough that the error estimate for the step is
1911 ! credible. The reduction is required to be at least as much as the step
1912 ! control parameter RS1. It is necessary to limit the reduction of HTRY
1913 ! at any one time because we may be misled in the size of the reduction
1914 ! that is appropriate due to nonlinearity of the differential equation
1915 ! and to inaccurate weights caused by HTRY much too large. The reduction
1916 ! is not permitted to be more than the step control parameter RS4.
1917 !
1918  cutbak = .false.
1919  if (argdif > comm%round_off*max(ynrm,ystgnm)) then
1920  if ((abs(htry)*fdiff) > (comm%stability_radius*argdif)) then
1921  scl = max(comm%rs4,min(comm%rs1, &
1922  (comm%stability_radius*argdif)/(abs(htry)*fdiff)))
1923  htry = scl*htry; cutbak = .true.
1924  end if
1925  end if
1926 !
1927  end subroutine stepa
1928 !
1929  subroutine stepb
1930 !
1931  real(kind=wp), dimension(:), pointer :: e !real!
1932 !
1933  e => comm%e
1934 !
1935  if (main) then
1936  err = maxval( abs( e(1)*yp + e(3)*stages(:,ptr(3)) + & !spec-ar!
1937  e(4)*stages(:,ptr(4)) + e(5)*stages(:,ptr(5)) + &
1938  e(6)*stages(:,ptr(6)) ) / &
1939  max( half*(abs(y)+abs(y_new)), thresh ) )
1940  else
1941  err = maxval( abs( e(1)*yp + e(3)*stages(:,ptr(3)) + & !spec-ar!
1942  e(4)*stages(:,ptr(4)) + e(5)*stages(:,ptr(5)) + &
1943  e(6)*stages(:,ptr(6)) ) / weights )
1944  end if
1945 !
1946  err = abs(comm%h)*err
1947 !
1948  end subroutine stepb
1949 !
1950 end subroutine step_r1

1953 subroutine stiff_r1(comm,f,toomch,sure_stiff)

2886 subroutine extra_stages(f,ytemp,xstage)
2887 !
2888 real(kind=wp), dimension(:), intent(out) :: ytemp, xstage !dep!

2890 interface
2891  function f(t,y)

2893  real(kind=wp), intent(in) :: t !indep!

2900 real(kind=wp), dimension(:), pointer :: yp, y_old, yp_old !dep!
2901 !
2902 real(kind=wp), dimension(:,:), pointer :: a !real!

2904 real(kind=wp), pointer :: h_old, t_old !indep!
2905 !
2906 integer :: i, j
2907 !
2908 a => comm%a
2909 stages => comm%stages
2910 c => comm%c
2911 yp => comm%yp
2912 y_old => comm%y_old
2913 yp_old => comm%yp_old
2914 h_old => comm%h_old
2915 t_old => comm%t_old
2916 !
2917 ! Compute the extra stages needed for interpolation using the facts that
2918 ! 1. Stage 1 is YP_OLD.
2919 ! 2. Stage i (i>1) is stored in STAGES(...,i-1).
2920 ! 3. This pair is FSAL, i.e. STAGES(...,7)=YP, which frees
2921 ! up STAGES(...,7) for use by stage 9.
2922 ! 4. XSTAGE is used for stage 10.
2923 ! 5. The coefficient of stage 2 in the interpolant is always 0, so
2924 ! STAGES(...,1) is used for stage 11.
2925 !
2926 do i = 9, 11
2927  do j = 1, i-1
2928  select case (j)
2929  case(1); ytemp = a(i,1)*yp_old
2930 ! could have used matmul here but that prevents increasing rank of dep-var
2931  case(2:7);ytemp = ytemp + a(i,j)*stages(:,j-1)
2932  case(8); ytemp = ytemp + a(i,j)*yp
2933  case(9); ytemp = ytemp + a(i,j)*stages(:,7)
2934  case(10); ytemp = ytemp + a(i,j)*xstage
2935  end select
2936  end do

2939  case(9)
2940  stages(:,7) = f(t_old+c(i)*h_old,ytemp)
2941  case(10)
2942  xstage = f(t_old+c(i)*h_old,ytemp)
2943  case(11)
2944  stages(:,1) = f(t_old+c(i)*h_old,ytemp)
2945  end select
2946  comm%f_count = comm%f_count + 1
2947 end do
