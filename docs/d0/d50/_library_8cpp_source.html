636  // object

638  Require( newBurnups.size() > 0 );

642  std::sort( newBurnups.begin(), newBurnups.end() );
643 
644  size_t numBU = transition_coeff_size(); // Number of burnups in library
645  std::vector<float> burnupOld; // list of burnups in old library
646 
647  std::vector<GridView_TransitionCoeff>
648  tCoeffViewVec; // viewer for tcoeff objects
649  tCoeffViewVec.reserve( numBU );

654  SCP_TransitionCoeff tcCopy = scp_transition_coeff_at( position );
655 
656  // make list of old burnups to interp to the new ones
657  burnupOld.push_back( tcCopy->burnup() );
658

660  tCoeffViewVec.emplace_back( tcCopy );
661  }
662  Check( increasing( burnupOld ) );
663 
664  // Set up Grid data
665  GD_t gdObj;
666 
667  GD_t::Vec_I extents = {numBU};
668  GD_t::Vec_Xptr grid = {&burnupOld[0]};
669 
670  gdObj.setx( extents, grid );
671 
672  // Setup responses for locations in Grid Data gdObj
673  GD_t::I_t numR = tCoeffViewVec[0].nr();

679  const GD_t::Vec_Rptr& resp = v.r();

681  for( size_t r = 0; r < numR; r++ ) responses.push_back( resp[r] );
682  }

684  // put responses into grid data at locations
685  gdObj.setr( tCoeffViewVec[0].rextents(), responses );

688  for( auto& bu : newBurnups )
689  {
690  GD_t::Vec_X wgts;
691 
692  // make weights for new burnup #bu
693  b_interp_func( burnupOld, bu, &wgts );

695 
696  // make new transition object

698  SP_TransitionCoeff tempTCoeff = tCoeffViewVec[0].permute( z );
699  // with "nearest" interpolator this is not true

701

703  tempTCoeff->set_burnup( bu );
704 
705  // put transition coefficient into the library
706  libNew->b_transition_coeff->push_back( tempTCoeff );
707  }
708 
709  // copy data from libOld that doesn't change (shallow copy for now)

718 SP_Library Library::interpolate_Interp1D( std::vector<double> newBurnups )
719 {
720  Require( newBurnups.size() > 0 );
721  using ScaleUtils::Math::Numeric::Interp1D_CubicSpline_SF;
722 
723  // Require(interpolator != nullptr);
724 
725  // Ensure burnups are increasing order
726  std::sort( newBurnups.begin(), newBurnups.end() );
727 
728  size_t numBU = transition_coeff_size(); // Number of burnups in library
729  std::vector<double> burnupOld; // interpolation x-values
730  std::vector<std::vector<double>> tCoeffData; // interpolation y-values
731  std::vector<double> tmpVec;
732 
733  // Push Transition coeffs into a vector of viewers
734  for( size_t position = 0; position < numBU; ++position )
735  {
736  SCP_TransitionCoeff tcCopy = scp_transition_coeff_at( position );
737 
738  // make list of old burnups to interp to the new ones
739  burnupOld.push_back( tcCopy->burnup() );
740 
741  // In terms of the interpolation, the burnups will be the x-values
742  // and the rest of the data will be the vectors of y-data.
743  // Constructing that y-data now out of all the TC data.
744  for( auto& matrix : *( tcCopy->scp_matrix() ) )
745  tmpVec.push_back( matrix );
746  for( auto& loss_xs : *( tcCopy->scp_loss_xs() ) )
747  tmpVec.push_back( loss_xs );
748  for( auto& fission_xs : *( tcCopy->scp_fission_xs() ) )
749  tmpVec.push_back( fission_xs );
750  for( auto& neutron_yields : *( tcCopy->scp_neutron_yields() ) )
751  tmpVec.push_back( neutron_yields );
752  for( auto& flux : *( tcCopy->scp_flux() ) ) tmpVec.push_back( flux );
753  if( tcCopy->scp_kappa_capture() != nullptr )
754  {
755  for( auto& kappa_capture : *( tcCopy->scp_kappa_capture() ) )
756  tmpVec.push_back( kappa_capture );
757  }
758  if( tcCopy->scp_kappa_fission() != nullptr )
759  {
760  for( auto& kappa_fission : *( tcCopy->scp_kappa_fission() ) )
761  tmpVec.push_back( kappa_fission );
762  }
763  tCoeffData.push_back( tmpVec );
764  tmpVec.resize( 0 );
765  }
766  Check( increasing( burnupOld ) );
767 
768  // Interpolate the TransitionCoeff data on the new burnup grid
769  Interp1D_CubicSpline_SF<double> interpolator;
770  std::vector<std::vector<double>> newTCs =
771  interpolator.interp_multi_list( newBurnups, burnupOld, tCoeffData );
772 
773  SP_Library libNew( new Library() );
774  SCP_TransitionCoeff oldTC = this->scp_transition_coeff_at( 0 );
775 
776  for( size_t bu = 0; bu < newBurnups.size(); ++bu )
777  {
778  SP_TransitionCoeff tempTC = SP_TransitionCoeff( new TransitionCoeff() );
779 
780  size_t start = 0;
781  size_t stop = oldTC->matrix_size();
782  std::vector<float> tmpMatrix( newTCs[bu].begin(),
783  newTCs[bu].begin() + stop );
784  tempTC->set_matrix( tmpMatrix );
785  Check( tempTC->matrix_size() == oldTC->matrix_size() );
786 
787  start = stop;
788  stop += oldTC->loss_xs_size();
789  std::vector<float> tmpLossXS( newTCs[bu].begin() + start,
790  newTCs[bu].begin() + stop );
791  tempTC->set_loss_xs( tmpLossXS );
792  Check( tempTC->loss_xs_size() == oldTC->loss_xs_size() );
793 
794  start = stop;
795  stop += oldTC->fission_xs_size();
796  std::vector<float> tmpFissionXS( newTCs[bu].begin() + start,
797  newTCs[bu].begin() + stop );
798  tempTC->set_fission_xs( tmpFissionXS );
799  Check( tempTC->fission_xs_size() == oldTC->fission_xs_size() );
800 
801  start = stop;
802  stop += oldTC->neutron_yields_size();
803  std::vector<float> tmpNeutronYields( newTCs[bu].begin() + start,
804  newTCs[bu].begin() + stop );
805  tempTC->set_neutron_yields( tmpNeutronYields );
806  Check( tempTC->neutron_yields_size() == oldTC->neutron_yields_size() );
807 
808  start = stop;
809  stop += oldTC->flux_size();
810  std::vector<float> tmpFlux( newTCs[bu].begin() + start,
811  newTCs[bu].begin() + stop );
812  tempTC->set_flux( tmpFlux );
813  Check( tempTC->flux_size() == oldTC->flux_size() );
814 
815  if( oldTC->scp_kappa_capture() != nullptr )
816  {
817  start = stop;
818  stop += oldTC->kappa_capture_size();
819 
820  std::vector<float> tmpKC( newTCs[bu].begin() + start,
821  newTCs[bu].begin() + stop );
822  tempTC->set_kappa_capture( tmpKC );
823  Check( tempTC->kappa_capture_size() ==
824  oldTC->kappa_capture_size() );
825  }
826 
827  if( oldTC->scp_kappa_fission() != nullptr )
828  {
829  start = stop;
830  stop += oldTC->kappa_fission_size();
831 
832  std::vector<float> tmpKF( newTCs[bu].begin() + start,
833  newTCs[bu].begin() + stop );
834  tempTC->set_kappa_fission( tmpKF );
835  Check( tempTC->kappa_fission_size() ==
836  oldTC->kappa_fission_size() );
837  }
838  tempTC->set_burnup( newBurnups[bu] );
839  // put transition coefficient into the library
840  libNew->b_transition_coeff->push_back( tempTC );
841  }
842 
843  // copy data from libOld that doesn't change (shallow copy for now)
844  libNew->b_definition = b_definition;
845  libNew->b_tag_manager = b_tag_manager;
846  libNew->b_decay_data = b_decay_data;
847  libNew->b_transition_structure = b_transition_structure;
848 
849  return ( libNew );
