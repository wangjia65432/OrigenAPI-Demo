167 // integral computation used internally
168 SP_Vec_Dbl Spectrum::computeUnionWts_1( const Vec_Dbl& union_grid,
169  const Vec_Int& union_to_old,
170  const Vec_Dbl& old_flux )
171 {
172  jDebugTaggedLine( "pass 1: integrate numerator" );
173  Vec_Dbl* c = new Vec_Dbl( union_grid.size() - 1, 0.0 );
174  for( size_t i = 0; i < union_grid.size() - 1; ++i )
175  {
176  ( *c )[i] += union_grid[i] - union_grid[i + 1];
177  }
178 
179  jDebugTaggedLine( "pass 2: integrate denominator" );
180  Vec_Dbl norm( old_flux.size(), 0.0 );
181  for( size_t i = 0; i < union_grid.size() - 1; ++i )
182  {
183  int io = union_to_old[i];
184  norm[io] += ( *c )[i];
185  }
186 
187  jDebugTaggedLine( "pass 3: normalize" );
188  for( size_t i = 0; i < union_grid.size() - 1; ++i )
189  {
190  int io = union_to_old[i];
191  ( *c )[i] /= norm[io];
192  }
193 
194  return SP_Vec_Dbl( c );
195 }
