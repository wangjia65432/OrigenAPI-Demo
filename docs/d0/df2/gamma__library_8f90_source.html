565  open(unit_old,file=trim(fileprefix)//".continuum.old",&
566  status=trim(dump_status),position=trim(dump_position))
567 
568  unit_new_de=1006
569  open(unit_new_de,file=trim(fileprefix)//".continuum.new.de",&
570  status=trim(dump_status),position=trim(dump_position))
571 
572  unit_new=1007
573  open(unit_new,file=trim(fileprefix)//".continuum.new",&
574  status=trim(dump_status),position=trim(dump_position))
575 
576 #endif
577 
578  do ! loop through the library
579 
580  !initialize lost stuff
581  unbinned_particles = 0.d0
582  unbinned_energy = 0.0d0
583 
584  ! read and identify a nuclide
585  ! Read header information as reals for format compatibility between libraries and compilers (i.e., GNU Fortran)
586  read (unit=gunit, fmt=*, iostat=iostatus, iomsg=msg ) nuc_id, nl, nl_x, nl_g, nl_c
587  nlines = floor(nl)
588  nlines_x = floor(nl_x)
589  nlines_g = floor(nl_g)
590  nlines_c = floor(nl_c)
591 
592  if( nlines<=0) cycle
593  if(iostatus < 0) exit ! Assume we're done reading the file
594  if(iostatus > 0) then
595  ! Error reading the file, notify user
596  scalereporterror(1, '(a,a)') "Error while processing gamma library: ", trim(full_name)
597  scalereporterror(1, '(a)') msg ; scalefinalizeerror()
598  exit
599  end if
600 
601  !initialize for this nuclide
602  binned_spect = 0.0d0
603 
604  write(rbuffer,'(a,5i8)') "Processing gamma nuclide", nuc_id, nlines, nlines_x, nlines_g, nlines_c
605  jdebugline(trim(rbuffer))
606 
607  !the +1 on ree is for the extra energy for the last bound
608  allocate(ree(nlines+1), rii(nlines))
609  read (unit=gunit, fmt="(10x,6es11.4)") (ree(i), rii(i), i=1, nlines)
610  jdebugline("Line data read.")
611 
612  ! establish the number of discrete and continuum lines
613  if (expand_continuum .and. nlines_c > 1) then
614  nl_discrete = nlines - nlines_c
615  nl_continuum = nlines_c
616  else
617  nl_discrete = nlines
618  nl_continuum = 0
619  end if
620 
621  ! read and bin all the lines
622  nbuf = 0
623  nuc = min(this%lib%find_nuclide(1, nuc_id), this%lib%find_nuclide(2, nuc_id), this%lib%find_nuclide(3, nuc_id) )
624  if (nuc>this%lib%get_itot()) then
625  write(rbuffer,'(a,i8,a)') "In the gamma library, nuclide with ID ", nuc_id, " was not found in the ORIGEN library."
626  jdebugline(trim(rbuffer))
627  deallocate(ree, rii)
628  cycle
629  end if
630 
631  do i = 1, nl_discrete
632  re = ree(i)
633  ri = rii(i)
634  nbuf = nbuf - 1
635  ! we have the line, now bin it
636  did_bin = .false.
637  do g = 1, this%opts%ngroups
638  ! if the line inside the group
639  if (real(re,4) > real(this%opts%bounds(g+1),4) .and. real(re,4) <= real(this%opts%bounds(g),4)) then
640  ! get the relative position within the group
641  f = (re - this%opts%bounds(g+1)) / (this%opts%bounds(g) - this%opts%bounds(g+1))
642  ! determine alternate binning
643  g_ = g
644  if (this%opts%split_near_boundary .and. f >= 0.97d0 .and. g > 1) g_ = g - 1
645  if (this%opts%split_near_boundary .and. f <= 0.03d0 .and. g < this%opts%ngroups) g_ = g + 1
646  ! if binning alternately, split the line intensity
647  if (g_ /= g) then
648  ri = ri / 2.
649  if (this%opts%conserve_line_energy) then
650  binned_spect(g_) = binned_spect(g_) + ri * re / this%e_gamma(g_)
651  else
652  binned_spect(g_) = binned_spect(g_) + ri
653  end if
654  end if
655  ! bin into the main group
656  if (this%opts%conserve_line_energy) then
657  binned_spect(g) = binned_spect(g) + ri * re / this%e_gamma(g)
658  else
659  binned_spect(g) = binned_spect(g) + ri
660  end if
661  ! flag as found
662  did_bin = .true.
663  ! stop searching
664  exit
665  end if
666  end do
667 
668  ! print unbinned photon warning
669  if ( .not. did_bin )then
670  unbinned_particles = unbinned_particles + ri
671  unbinned_energy = unbinned_energy + re*ri
672  if ( this%opts%print_unbinned) then
673  write (rbuffer, fmt="(' photon could not be binned; id= ',i7,' ; energy= ',e11.4, ' ; intensity=',e11.4,' energies = ',e11.4,' .. ',e11.4)") &
674  nuc_id, real(re,4), real(ri,4), this%opts%bounds(1), &
675  this%opts%bounds(this%opts%ngroups+1)
676  call output(trim(rbuffer))
677  end if
678  end if
679  end do
680 
681 
682  !! process the continuum "lines"
683  if( nl_continuum>0 )then
684 
685  !initialize for this nuclide
686  binned_continuum_spect=0.0d0
687 
688  !NOTE: this is a hack but we need to
689  !add the final energy value is based on the file type
690  !the next to last energy is:
691  e_next_to_last = ree(nl_discrete+nl_continuum)
692 
693  !the last bin width we can either calculated from the next to last bin
694  !if we have the data points (2 or more) or we just use the energy
695  !given as the delta-E
696  if( nl_continuum>1 )then
697  de = ree(nl_discrete+nl_continuum)-ree(nl_discrete+nl_continuum-1);
698  else
699  de = ree(nl_discrete+nl_continuum)-0.d0
700  endif
701 
702  !For bremsstrahlung we can try to get the next bin because we know
703  !the group structure.
704  if( filename(1:4)=="mpbr" )then
705  gindex=1
706  e_last = next_larger_brem_continuum_energy(gindex,e_next_to_last)
707 
708  !failed to get the next larger and we fall back on the calculated
709  !de above
710  if( gindex==0 )then
711  e_last = e_next_to_last + de
712  end if
713  ree(nl_discrete+nl_continuum+1)=e_last
714 
715  !For other stuff, we just assume the next bin is the same
716  !width as the previous bin, and therefore the last energy
717  !is E[N] = E[N-1] + (E[N-1]-E[N-2])
718  else
719  ree(nl_discrete+nl_continuum+1) = e_next_to_last + de
720  endif
721 
722 #ifdef ORIGEN_DUMP_SPECTRA
723 
724  !dump the spectra
725  call dump_orig_spectra_de(filename,unit_orig_de,&
726  ree(nl_discrete+1:nl_discrete+nl_continuum+1),&
727  rii(nl_discrete+1:nl_discrete+nl_continuum),nuc_id)
728  call dump_orig_spectra(filename,unit_orig,&
729  ree(nl_discrete+1:nl_discrete+nl_continuum+1),&
730  rii(nl_discrete+1:nl_discrete+nl_continuum),nuc_id)
731 
732 #endif
733 
734  !this does the actual rebinning
735  call this%rebin_continuum(filename,&
736  ree(nl_discrete+1:nl_discrete+nl_continuum+1),&
737  rii(nl_discrete+1:nl_discrete+nl_continuum),&
738  binned_continuum_spect,nuc_id,&
739  unbinned_particles,&
740  unbinned_energy)
741 
742 #ifdef ORIGEN_DUMP_SPECTRA
743  call dump_binned_spectra_de(filename,unit_new_de,this%opts%bounds,binned_continuum_spect,nuc_id)
744  call dump_binned_spectra(filename,unit_new,this%opts%bounds,binned_continuum_spect,nuc_id)
745 #endif
746 
747  binned_spect = binned_spect + binned_continuum_spect
748 
749  end if
750 
751 
752  call this%assign_spectra( nuc_id, binned_spect, unbinned_particles, unbinned_energy )
753 
754  deallocate(ree, rii)
755 
756  end do !nuclide loop
757 
758 !
759 !Create new unit test insists for tstGamma.f90
760 !
761 !these are written to Insists.#.txt where # is the
762 !pass number for read_photon_library. You generally
763 !want to use the last pass for comparisons. Just copy and paste
764 !to tstGamma.f90:ogc_check
765 #ifdef ORIGEN_CREATE_NEW_GAMMA_INSISTS
766  write(insists_filename,'(a,i1,a)')"Insists.",pass_number,".txt"
767  open(unit=1001,file=insists_filename)
768  do nuc = 1, this%lib%get_itot()
769  write(1001,*)'Insist(this%has_gamma(',nuc,.eqv.')',merge(".true. ",".false.",this%has_gamma(nuc)),',"',this%lib%nucl(nuc),' has_gamma")'
770  if( this%has_gamma(nuc) ) then
771  write(1001,*)'ref=',sum(this%binned(:,nuc))
772  write(1001,*)'new=sum(this%binned(:,',nuc,'))'
773  write(1001,*)'write(msg,"(a,es18.6,a)")" total counts are conserved (ref=',sum(this%binned(:,nuc)),', new=",new,")"'
774  write(1001,*)'Insist(abs((1.d-99+new)/(1.d-99+ref)-1.d0)<1d-6,"',this%lib%nucl(nuc),'"//trim(msg))'
775  else
776  write(1001,*)'Insist(sum(this%binned(:,',nuc,'))==',0.d0,',"',this%lib%nucl(nuc),' has zero counts if not has_gamma")'
777  endif
778  enddo
779  close(1001)
780 #endif
781 
782 !close units for dumped spectra
783 #ifdef ORIGEN_DUMP_SPECTRA
784  close(unit_orig)
785  close(unit_orig_de)
786  close(unit_new)
787  close(unit_new_de)
788  close(unit_old)
789  close(unit_old_de)
790 #endif
791 
792  close(gunit)
793 
794  end subroutine gamma_library__read_photon_library

925 subroutine gamma_library__rebin_continuum(this,filename,xvalues,yvalues,binned_spectra,nuc_id,&
926  unbinned_particles,unbinned_energy)
927  use scaledata_energyunionizer_m
928  use scaledata_mgfluxmapper_m
929  use scalestl_vec_dbl_m
930  use scalestl_vec_int_m
931 
932  class(gamma_library_class) :: this
933  character(*),intent(in) :: filename
934  real,intent(in) :: xvalues(:)
935  real,intent(in) :: yvalues(:)
936 
937  real(C_DOUBLE), intent(inout) :: binned_spectra(this%opts%ngroups)
938  real(C_DOUBLE), intent(inout) :: unbinned_particles,unbinned_energy
939  integer,intent(in) :: nuc_id
940 
941  type(scalestl_vec_dbl) :: old_grid,new_grid,union_grid,e_integrals
942  type(scalestl_vec_dbl) :: old_flux,new_flux
943  type(scaledata_energyunionizer) :: sd_eu
944  type(scaledata_mgfluxmapper) :: sd_mgfm
945  logical :: need_to_apply_max_offset
946 
947  integer :: g,ng_old
948  integer(C_SIZE_T) :: guser,end_guser
949  integer :: gindex
950  character(10000) :: mystr
951  real(C_DOUBLE) :: e,bare,bars,dele,last_flux,next_dele
952  real(C_DOUBLE) :: e_old_min,e_old_max
953  real(C_DOUBLE) :: e_new_min,e_new_max
954  real(C_DOUBLE) :: e_user_min,e_user_max
955 
956  ng_old = size(yvalues)
957 
958  !!!0SErrLine(1,*)'rebin_continuum, nuc_id=',nuc_id," ng_old=",ng_old
959  jdebugline("entering rebin_continuum.")
960 
961  !quick return if no xvalues
962  if( ng_old==0 )then
963 ! do g=1,this%opts%ngroups
964 ! !!0SErrLine(1,*)'#%# ',this%opts%bounds(g),binned_spectra(g)
965 ! end do
966  return
967  end if
968 
969  !!0SErrLine(1,*)'INPUT GRID:'
970  !!0do g=1,ng_old+1
971  !!0SErrLine(1,*)g,xvalues(g)
972  !!0end do
973 
974  e_old_max = real(xvalues(ng_old+1),C_DOUBLE)
975  e_old_min = real(xvalues(1),C_DOUBLE)
976  !!0SErrLine(1,*)'e_old_min=',e_old_min
977  !!0SErrLine(1,*)'e_old_max=',e_old_max
978 
979  e_user_max = this%opts%bounds(1);
980  e_user_min = this%opts%bounds(this%opts%ngroups+1)
981  !!0SErrLine(1,*)'e_user_min=',e_user_min
982  !!0SErrLine(1,*)'e_user_max=',e_user_max
983 
984  !new data is already from high to low energy
985  !we will sort in the canonicalize at the end so it doesn't matter
986  !that we are adding these out of order or adding multiples
987  call new_grid%initialize()
988  call new_flux%initialize()
989  need_to_apply_max_offset = .false.
990  if( approx_gt(e_old_max,e_user_max,bin_boundary_match_tol) )then
991  need_to_apply_max_offset = .true.
992  call new_grid % push_back(e_old_max)
993  end if
994  if( approx_lt(e_old_min,e_user_min,bin_boundary_match_tol) )then
995  call new_grid % push_back(e_old_min)
996  end if
997 
998  do g=1,this%opts%ngroups+1
999  e=this%opts%bounds(g)
1000  call new_grid % push_back(e)
1001  end do
1002 
1003  call sd_eu % initialize()
1004  call sd_eu % canonicalize(new_grid)
1005  e_new_min = new_grid%back()
1006  e_new_max = new_grid%front()
1007  !!0SErrLine(1,*)'e_new_min=',e_new_min
1008  !!0SErrLine(1,*)'e_new_max=',e_new_max
1009  !!0SErrLine(1,*)'NEW GRID:'
1010  !!0do g=1,new_grid%size()
1011  !!0SErrLine(1,*)g,new_grid%at(g)
1012  !!0end do
1013  jdebugline("done setting up new grid.")
1014 
1015 
1016  !need to reverse the xvalues
1017  !and make sure we have maximum energy FOR THIS DATA SET
1018  call old_grid%initialize()
1019  call old_flux%initialize()
1020 
1021  !make sure we have the same end points
1022  if( approx_gt(e_new_max,e_old_max,bin_boundary_match_tol) )then
1023  call old_grid%push_back(e_new_max)
1024  call old_flux%push_back(0.d0)
1025  end if
1026 
1027  !add the other grid lines in the reverse order
1028  do g=ng_old,1,-1
1029  call old_flux % push_back(real(yvalues(g),C_DOUBLE))
1030  call old_grid % push_back(real(xvalues(g+1),C_DOUBLE))
1031  end do
1032  call old_grid % push_back(real(xvalues(1),C_DOUBLE))
1033 
1034  if( approx_lt(e_new_min,e_old_min,bin_boundary_match_tol) )then
1035  call old_grid%push_back(e_new_min)
1036  call old_flux%push_back(0.d0)
1037  end if
1038 
1039  !!0SErrLine(1,*)'OLD GRID:'
1040  !!0do g=1,old_grid%size()
1041  !!0SErrLine(1,*)g,old_grid%at(g)
1042  !!0end do
1043  jdebugline("done setting up old grid.")
1044 
1045  call sd_eu%setGrids(old_grid,new_grid,bin_boundary_match_tol)
1046 
1047  call sd_eu%unionGrid(union_grid)
1048 
1049  call sd_mgfm % initialize()
1050  call sd_mgfm % fluxIntegrals_1(e_integrals,union_grid)
1051  call sd_mgfm % setUnionizer(sd_eu,e_integrals)
1052 
1053  call new_flux%resize(new_grid%size()-1)
1054  call sd_mgfm % mapFlux(old_flux,new_flux)
1055  jdebugline("done mapping flux.")
1056 
1057 !!!0SErrLine(1,*)'nuc_id=',nuc_id,' old_flux%size=',old_flux%size()
1058 !!!0SErrLine(1,*)'nuc_id=',nuc_id,' xvalues%size=',size(xvalues)
1059 !!!0SErrLine(1,*)'nuc_id=',nuc_id,' new_flux%size=',new_flux%size()
1060 
1061  !guser is the group index in the user-group structure
1062  !because we added some groups we need to offset
1063  guser=1
1064  if( need_to_apply_max_offset )then
1065  do guser=1,new_grid%size()
1066  if( approx_eq(new_grid%at(guser),this%opts%bounds(1),bin_boundary_match_tol) )exit;
1067  !accumulate lost particles/energy in the high energy groups
1068  unbinned_particles = unbinned_particles + new_flux%at(guser)
1069  unbinned_energy = unbinned_energy + 0.5*new_flux%at(guser)*( new_grid%at(guser) + new_grid%at(guser+1) )
1070  end do
1071  end if
1072 
1073  do g = 1, this%opts%ngroups
1074  check(guser<=new_flux%size())
1075  !!0SErrLine(1,*)'g',g,this%opts%bounds(g),new_grid%at(guser)
1076  !!0SErrLine(1,*)'g+1',g+1,this%opts%bounds(g+1),new_grid%at(guser+1)
1077  check(approx_eq(this%opts%bounds(g),new_grid%at(guser),bin_boundary_match_tol))
1078  check(approx_eq(this%opts%bounds(g+1),new_grid%at(guser+1),bin_boundary_match_tol))
1079  binned_spectra(g) = binned_spectra(g) + new_flux%at(guser)
1080  guser=guser+1
1081  end do
1082 
1083  !iterate over low energy groups to get lost particles/energy
1084  end_guser = guser
1085  do guser=end_guser,new_grid%size()-1
1086  unbinned_particles = unbinned_particles + new_flux%at(guser)
1087  unbinned_energy = unbinned_energy + 0.5*new_flux%at(guser)*( new_grid%at(guser) + new_grid%at(guser+1) )
1088  end do
1089 
1090  call sd_eu%destroy()
1091  call sd_mgfm%destroy()
1092  call old_grid%destroy()
1093  call old_flux%destroy()
1094  call new_grid%destroy()
1095  call new_flux%destroy()
1096 
1097 end subroutine gamma_library__rebin_continuum
