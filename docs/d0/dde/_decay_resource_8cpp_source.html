90 void DecayResource::calculate_decay_constants()
91 {
92  // This computation is emulating the
93  // Fortran source from src/couple/halfl.f90
94  // data c/ 6.9315E-01, 1.1552E-02, 1.9254E-04, 8.0226E-06, 2.1965E-08, 0.0,
95  // &
96  // 2.1965E-11, 2.1965E-14, 2.1965E-17/
97  // if (a <= 0.0) then
98  // if (i == 6) go to 110
99  // a = 9.99
100  // return
101  // endif
102  // a = c(i)/a
103  // return
104  // 110 continue
105  // a = 0.0
106  // sec,min,hr,day,yr,NONE,1e3 yr, 1e6 yr, 1e9 yr
107 
108  double conversion_data[] = {6.9315E-01,
109  1.1552E-02,
110  1.9254E-04,
111  8.0226E-06,
112  2.1965E-08,
113  0.0,
114  2.1965E-11,
115  2.1965E-14,
116  2.1965E-17};
117 
118  std::map<int, DecayParent>::iterator it;
119 
120  for( it = parent_map.begin(); it != parent_map.end(); it++ )
121  {
122  double decay_constant = 0.0;
123  double half_life = it->second.get_half_life();
124 
125  int iu = it->second.get_time_units();
126  if( half_life <= 0.0 )
127  {
128  if( iu == 6 )
129  {
130  decay_constant = 0.0;
131  }
132  else
133  {
134  decay_constant = 9.99;
135  }
136  }
137  else
138  {
139  Check( iu != 6 );
140  decay_constant = conversion_data[iu - 1] / half_life;
141  }
142  it->second.set_decay_constant( decay_constant );
143  }
144 }

289 {

291 
292  std::map<int, DecayParent>::const_iterator it;
293  for( it = parent_map.begin(); it != parent_map.end(); it++ )
294  {
295  int id = it->first;

297 
298  double recoverable_energy_value = parent.get_recoverable_energy_value();
299  double half_life = parent.get_half_life();
300  if( half_life > 0.0 && recoverable_energy_value == 0.0 )
301  {
302  std::stringstream ss;
303  ss << "Parent id=" << id << " cannot have a half_life=" << half_life
304  << ">0.0 and recoverable_energy_value==0.0.";

306  }
307 
308  std::map<int, DecayChannel>::const_iterator it2;
309  if( parent.get_const_channel_map().size() == 0 && half_life != 0.0 )
310  {
311  std::stringstream ss;
312  ss << "Parent id=" << id
313  << " cannot have a half_life>0.0 and no decay modes.";
314  errors.push_back( ss.str() );
315  }
316  double sum_branch_ratio = 0;
317  for( it2 = parent.get_const_channel_map().begin();
318  it2 != parent.get_const_channel_map().end();
319  it2++ )
320  {
321  int decay_mode = it2->first;
322  if( !ScaleData::Utils::is_valid_decay_mode( decay_mode ) )
323  {
324  std::stringstream ss;
325  ss << "Parent id=" << id
326  << " has invalide decay_mode=" << decay_mode << ".";
327  errors.push_back( ss.str() );
328  }
329  DecayChannel channel = it2->second;
330 
331  sum_branch_ratio += channel.get_branch_ratio();
332 
333  int yield_type = channel.get_yield_type();
334  std::map<int, double>::const_iterator it3;
335  double sum_yield = 0;
336  for( it3 = channel.get_const_yield_map().begin();
337  it3 != channel.get_const_yield_map().end();
338  it3++ )
339  {
340  int did = it3->first;
341  sum_yield += it3->second;
342  if( yield_type == ISOMERIC_YIELDS )
343  {
344  if( did < 0 && did > 9 )
345  {
346  std::stringstream ss;
347  ss << "Parent id=" << id << " decay_mode=" << decay_mode
348  << " has yield yield_type=ISOMERIC_YIELDS, which "
349  "means only the isomeric state is given (0-9) but"
350  << " I=" << did << " was found.";
351  errors.push_back( ss.str() );
352  }
353  }
354  else if( yield_type == IZZZAAA_YIELDS )
355  {
356  std::stringstream ss;
357  ss << "Parent id=" << id << " decay_mode=" << decay_mode
358  << " has yield yield_type=IZZZAAA_YIELDS, with IZZZAAA="
359  << did << ".";
360  this->check_id( id, ss.str(), errors );
361  }
362  else if( yield_type == MISSING_YIELDS )
363  {
364  std::stringstream ss;
365  ss << "Parent id=" << id << " decay_mode=" << decay_mode
366  << " has yield yield_type=MISSING_YIELDS which means no "
367  "yields may be present.";
368  errors.push_back( ss.str() );
369  }
370  else
371  {
372  std::stringstream ss;
373  ss << "Parent id=" << id << " decay_mode=" << decay_mode
374  << " has yield yield_type=UNKNOWN (" << yield_type
375  << ").";
376  errors.push_back( ss.str() );
377  }
378  }
379  }
380  if( parent.get_const_channel_map().size() > 0 &&
381  std::abs( sum_branch_ratio - 1.0 ) > 1.0e-5 )
382  {
383  std::stringstream ss;
384  ss << "The sum of branch ratios for id=" << id << " is "
385  << sum_branch_ratio << ". It should be closer to 1.";
386  errors.push_back( ss.str() );
387  }
388  }
389  // no errors were introduced
390  return errors.size() == nErr;
391 }
