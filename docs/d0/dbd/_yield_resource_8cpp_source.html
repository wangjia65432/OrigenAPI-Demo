175 void YieldParent::get_interp_yields( Vec_Dbl* interpolated_yields,
176  double energy,
177  int interp ) const
178 {
179  size_t nsets = this->num_energies();
180  Check( nsets >= 1 && nsets <= 4 );
181  if( nsets == 1 )
182  {
183  *interpolated_yields = data_at( 0 ).yields();
184  }
185  else if( nsets == 2 || nsets == 3 || nsets == 4 )
186  {
187  // handle extrapolation
188  if( energy <= data_at( 0 ).energy() )
189  {
190  *interpolated_yields = Vec_Dbl( data_at( 0 ).yields() );
191  }
192  else if( energy > data_at( nsets - 1 ).energy() )
193  {
194  *interpolated_yields = Vec_Dbl( data_at( nsets - 1 ).yields() );
195  }
196  else
197  {
198  interpolated_yields->resize( num_fps() );
199 
200  // interpolation
201  size_t ja = 0;
202  size_t jb = 1;
203  for( size_t iset = 0; iset < nsets - 1; ++iset )
204  {
205  if( energy > data_at( iset ).energy() &&
206  energy <= data_at( iset + 1 ).energy() )
207  break;
208  ja++;
209  jb++;
210  }
211 
212  double ea = data_at( ja ).energy();
213  double eb = data_at( jb ).energy();
214 
215  const YieldData& dataA = this->data_at( ja );
216  const YieldData& dataB = this->data_at( jb );
217 
218  size_t fp_size = fp_map().size();
219  Check( fp_size == dataA.yields().size() );
220  Check( fp_size == dataB.yields().size() );
221 
222  for( size_t i = 0; i < fp_size; ++i )
223  {
224  double yield = endf::Tab1::interpolate( ea,
225  dataA.yield_at( i ),
226  eb,
227  dataB.yield_at( i ),
228  energy,
229  interp );
230  ( *interpolated_yields )[i] = yield;
231  }
232  }
233  }
234  else
235  {
236  Insist( false,
237  "Fatal error: number of fission yield sets must equal 1-4." );
238  }

325 bool YieldResource::check_consistency( Vec_Str& errors ) const
326 {
327  size_t nErr = errors.size();
328 
329  auto ite = b_fissionable_map.end();
330  for( auto it = b_fissionable_map.begin(); it != ite; ++it )
331  {
332  int id = it->first;
333  const YieldParent& parent = b_fissionables[it->second];
334  if( parent.id() != id )
335  {
336  errors.push_back(
337  nprintf( "Resource fissionable id=%d is "
338  "inconsistent with the parent id=%d\n",
339  parent.id(),
340  id ) );
341  }
342  for( size_t i = 0; i < parent.data().size(); i++ )
343  {
344  const YieldData& data = parent.data()[i];
345 
346  if( parent.fp_map().size() != data.yields().size() )
347  {
348  std::stringstream ss;
349  ss << "Parent id=" << id
350  << " has inconsistent yield data size. NFPs="
351  << parent.fp_map().size() << " but yield set i=" << i + 1
352  << " has NFPs=" << data.yields().size();
353  errors.push_back( ss.str() );
354  }
355  double sum = 0;
356  FPMap_t::const_iterator fp_it;
357  for( fp_it = parent.fp_map().begin();
358  fp_it != parent.fp_map().end();
359  ++fp_it )
360  {
361  int fp_iza = fp_it->first;
362  size_t j = fp_it->second;
363  int a = fp_iza % ScaleData::Utils::PIZZZAAA_Z_OFFSET;
364  if( a > IMPLICIT_BYPRODUCT_MAX_MASSNUMBER )
365  {
366  sum += data.yields()[j];
367  }
368  }
369  if( std::abs( sum - 2.0 ) > YIELD_TOLERANCE )
370  {
371  std::stringstream ss;
372  ss << "Parent id=" << id
373  << " fission yields for set i=" << i + 1
374  << " do not meet tolerance requirement std::abs(sum-2.0)="
375  << std::abs( sum - 2.0 ) << ">" << YIELD_TOLERANCE << ".";
376  errors.push_back( ss.str() );
377  }
378 
379  Check( data.energy() > 0 );
380 
381  // check that energies are increasing
382  if( i > 0 )
383  {
384  if( data.energy() < parent.data()[i - 1].energy() )
385  {
386  std::stringstream ss;
387  ss << "Parent id=" << id
388  << " has lower fission yield energy for set i=" << i + 1
389  << "than previous set";
390  }
391  }
392  }
393  }
394 
395  // no errors were introduced
396  return errors.size() == nErr;
397 }
