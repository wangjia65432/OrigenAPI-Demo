92 // allocate step
93 void Material::allocate_step()
94 {
95  Require( is_initialized() );
96  jDebugLine( " adding step for material=" << this->name() );
97 
98  // set bos
99  b_trel_bos = 0.0;
100  b_amount_bos = b_point_amount_cm2barn.back();
101 
102  // make eos a copy of bos initially
103  b_trel_eos = 0.0;
104  b_amount_eos = std::make_shared<DoubleList>( *b_amount_bos );
105 
106  // clear mos
107  clear_mos();
108 
109  // add values
110  b_step_trx.push_back( nullptr );
111  b_point_time_s.push_back( 0.0 );
112  b_flux_parameter.push_back( UNKNOWN );
113  b_step_flux_cm2s.push_back( -1.0 );
114  b_step_power_watt.push_back( -1.0 );
115  b_point_amount_cm2barn.push_back( b_amount_eos );
116 
117  // reset histories
118  b_param_hist.clear();
119  b_flux_hist.clear();
120  b_power_hist.clear();
121 }

123 /*!
124  * \brief prepare to perform a new step with double the time
125  * \return
126  */
127 void Material::repeat_step_doubled()
128 {
129  double dt0 = dt();
130 
131  const bool wrong_way( true );
132  if( wrong_way )
133  {
134  // save end as middle
135  b_amount_mos = b_amount_eos;
136 
137  // flush newest step
138  flush_newest();
139 
140  // add step back
141  add_step( 2 * dt0 );
142  }
143  else
144  {
145  // double time
146  set_dt( 2 * dt0 );
147 
148  // set bos pointer
149  b_trel_bos = 0.0;
150  b_amount_bos = amount_bos();
151 
152  // set mos to previous final
153  b_trel_mos = 0.0;
154  b_amount_mos = b_amount_eos;
155 
156  // make eos a copy of mos
157  b_trel_eos = 0.0;
158  b_amount_eos = std::make_shared<DoubleList>( *b_amount_mos );
159  b_point_amount_cm2barn.back() = b_amount_eos;
160 
161  // reset histories
162  b_param_hist.clear();
163  b_flux_hist.clear();
164  b_power_hist.clear();
165  }

257 /*!
258  * \brief flush newest step of information (including times)
259  * \return
260  */
261 void Material::flush_newest()
262 {
263  // pop off everything
264  b_step_flux_cm2s.pop_back();
265  b_step_power_watt.pop_back();
266  b_step_trx.pop_back();
267  b_point_time_s.pop_back();
268  b_point_amount_cm2barn.pop_back();
269  b_flux_parameter.pop_back();
270 
271  // reset relative time pointers
272  b_trel_bos = 0.0;
273  b_trel_mos = 0.0;
274  b_trel_eos = 0.0;
275 
276  // reset amount pointers
277  b_amount_bos = b_point_amount_cm2barn.front();
278  b_amount_mos = nullptr;
279  b_amount_eos = nullptr;
280 
281  // re-associate amount pointers
282  size_t n = nsteps();
283  if( n > 0 )
284  {
285  b_amount_bos = b_point_amount_cm2barn[n - 1];
286  b_amount_eos = b_point_amount_cm2barn[n];
287  }
288 
289  // reset histories
290  b_param_hist.clear();
291  b_flux_hist.clear();
292  b_power_hist.clear();
293 }

949 double integrate_history( const Vec_Dbl& scaling,
950  const Material::History_t& hist )
951 {
952  Require( scaling.size() == hist.size() );
953  size_t i = 0;
954  double acc = 0;
955  double trel1, frel1, trel0, frel0;
956  double s0;
957  double s1;
958  for( auto& it : hist )
959  {
960  trel1 = it.first;
961  frel1 = it.second;
962  s1 = scaling[i];
963  if( i > 0 )
964  acc += 0.5 * ( trel1 - trel0 ) * ( s1 * frel1 + s0 * frel0 );
965  ++i;
966  trel0 = trel1;
967  frel0 = frel1;
968  s0 = s1;
969  }
970  return acc;
971 }

1294 void Material::solve( Vec_Dbl& dtrel, Vec_Dbl* flux, Vec_Dbl* power )
1295 {
1296  flux->resize( dtrel.size() );
1297  power->resize( dtrel.size() );
1298 
1299  double trel0 = 0.0;
1300  double trel1;
1301  bool done = false;
1302  for( size_t i = 0; i < dtrel.size(); ++i )
1303  {
1304  trel1 = trel0 + dtrel[i];
1305  done = solve_inplace(
1306  0, trel0, trel1, 1.0, 1.0, &( *flux )[i], &( *power )[i] );
1307  Check( done ? i == dtrel.size() - 1 : i < dtrel.size() - 1 );
1308  trel0 = trel1;
1309  }
1310 
1311  Ensure( ScaleSTL::approx_eq( 1.0, b_trel_eos ) );
1312 }
1313 //----------------------------------------------------------------------
1314 
1315 // if nlit==0
1316 // depletes from END-OF-STEP number densities and updates
1317 // them in place without altering the BEGIN-OF-STEP number densities
1318 //(used for substepping)
1319 // if nlit>0
1320 //(redoes the substep calculation)
1321 bool Material::solve_inplace( size_t nlit,
1322  double trel0,
1323  double trel1,
1324  double z0,
1325  double z1,
1326  double* flux,
1327  double* power )
1328 {
1329  Check( b_amount_eos.get() == b_point_amount_cm2barn.back().get() );
1330 
1331  // initialize swapping and set trels
1332  init_mos();
1333  if( nlit == 0 ) swap_mos_eos();
1334  // nlit==0 | nlit>0
1335  // slot mos : a(trel0) | a(trel-1)
1336  // slot bos : a(0)
1337  // slot eos : garbage
1338 
1339  swap_mos_bos();
1340  // nlit==0 | nlit>0
1341  // slot mos : a(0) |
1342  // slot bos : a(trel0) | a(trel-1)
1343  // slot eos : garbage
1344 
1345  b_trel_bos = trel0;
1346  b_trel_eos = trel1;
1347 
1348  // check
1349  Check( ScaleSTL::approx_eq( 0.0, b_trel_mos ) );
1350 
1351  // solve
1352  solve_kernel( trel0, trel1, z0, z1, flux, power );
1353  // slot eos : a(trel1)
1354 
1355  swap_mos_bos();
1356  // slot bos : a(0)
1357  // slot mos : a(trel0)
1358 
1359  // ensure we are leaving in the right state
1360  Ensure( ScaleSTL::approx_eq( 0.0, b_trel_bos ) );
1361  Ensure( ScaleSTL::approx_eq( trel0, b_trel_mos ) );
1362  Ensure( ScaleSTL::approx_eq( trel1, b_trel_eos ) );
1363 
1364  jDebugLine( " amounts[at*cm^2/barn]:"
1365  << " bos("
1366  << b_trel_bos
1367  << ")="
1368  << ScaleSTL::sum( *b_amount_bos )
1369  << " boss("
1370  << b_trel_mos
1371  << ")="
1372  << ScaleSTL::sum( *b_amount_mos )
1373  << " eoss("
1374  << b_trel_eos
1375  << ")="
1376  << ScaleSTL::sum( *b_amount_eos ) );
1377  Check( b_amount_eos.get() == b_point_amount_cm2barn.back().get() );
1378  return ScaleSTL::approx_eq( 1.0, b_trel_eos );
1379 }

1382 
1383 // solves the subinterval over relative time trel0 --> trel1 which are both
1384 // in [0,1] with scaling factor z0 and z1 for the flux or power
1385 // and returns the average flux and average power over the interval
1386 void Material::solve_kernel( double trel0,
1387  double trel1,
1388  double z0,
1389  double z1,
1390  double* flux,
1391  double* power )
1392 {
1393  Require( z0 > 0.0 );
1394  Require( z1 > 0.0 );
1395  // z0,z1 are scaling factors for values at trel0 and trel1
