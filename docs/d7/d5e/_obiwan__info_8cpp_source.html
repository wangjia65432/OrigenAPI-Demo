192 int Obiwan_info::execute( std::ostream& out, std::ostream& err )
193 {
194  // quick returns
195  if( b_opts.get( "help", false ) )

197  out << help() << std::endl;

202  out << b_opts.writeErrorStack();

209  << std::setw( 9 ) << "fileType" << std::setw( 15 ) << "file/sysEndian"
210  << std::setw( 11 ) << "appVersion" << std::setw( 6 ) << "ms"
211  << " "
212  << "directory" << std::endl
213  << std::flush;
214 
215  std::string sysEndian =
216  ScaleUtils::IO::EndianUtils::system_is_big_endian() ? "big" : "little";
217 
218  // Loop over files.
219  auto files = b_opts.get<StringList*>( "paths" );
220  for( auto path : *files )
221  {
222  // ORIGEN_MACRO_loadMulti requires:

228  ORIGEN_MACRO_loadMulti( info );
